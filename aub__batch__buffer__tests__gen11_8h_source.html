141 aubFile.writeMMIO(AubMemDump::computeRegisterOffset(mmioBase, 0x2514), contextDescriptor.ulData[1]);

149 AubMemDump::computeRegisterOffset(mmioBase, 0x2234), //EXECLIST_STATUS
150 0x00008000,
151 0x00008000,
152 false,
153 CmdServicesMemTraceRegisterPoll::TimeoutActionValues::Abort);

155 alignedFree(pRing);
156 alignedFree(pLRCABase);
157 alignedFree(pGlobalHWStatusPage);

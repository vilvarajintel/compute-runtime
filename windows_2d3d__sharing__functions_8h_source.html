161template <typename D3DSharing>
162static inline cl_int getSupportedDXTextureFormats(cl_context context, cl_mem_object_type imageType, cl_uint plane,
163 cl_uint numEntries, DXGI_FORMAT *formats, cl_uint *numImageFormats) {
164 Context *pContext = castToObject<Context>(context);
165 if (!pContext) {
166 return CL_INVALID_CONTEXT;
167 }
168
169 auto pSharing = pContext->getSharing<D3DSharingFunctions<D3DSharing>>();
170 if (!pSharing) {
171 return CL_INVALID_CONTEXT;
172 }
173
174 size_t numberOfFormats = 0;
175 if (plane <= 1) {
176 auto supported_formats = pSharing->retrieveTextureFormats(imageType, plane);
177 numberOfFormats = supported_formats.size();
178
179 if (formats != nullptr) {
180 memcpy_s(formats, sizeof(DXGI_FORMAT) * numEntries, supported_formats.data(), sizeof(DXGI_FORMAT) * std::min(static_cast<size_t>(numEntries), numberOfFormats));
181 }
182 }
183
184 if (numImageFormats) {
185 *numImageFormats = static_cast<cl_uint>(numberOfFormats);
186 }
187 return CL_SUCCESS;
188}

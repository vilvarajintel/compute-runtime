27 void handleQueryItem(drm_i915_query_item *queryItem) override {
28 switch (queryItem->query_id) {
29 case DRM_I915_QUERY_MEMORY_REGIONS:
30 if (queryMemoryRegionInfoSuccessCount == 0) {
31 queryItem->length = -EINVAL;
32 } else {
33 queryMemoryRegionInfoSuccessCount--;
34 auto numberOfLocalMemories = 1u;
35 auto numberOfRegions = 1u + numberOfLocalMemories;
36 int regionInfoSize = sizeof(PROD_DG1::drm_i915_query_memory_regions) + numberOfRegions * sizeof(PROD_DG1::drm_i915_memory_region_info);
37 if (queryItem->length == 0) {
38 queryItem->length = regionInfoSize;
39 } else {
40 EXPECT_EQ(regionInfoSize, queryItem->length);
41 auto queryMemoryRegionInfo = reinterpret_cast<PROD_DG1::drm_i915_query_memory_regions *>(queryItem->data_ptr);
42 EXPECT_EQ(0u, queryMemoryRegionInfo->num_regions);
43 queryMemoryRegionInfo->num_regions = numberOfRegions;
44 queryMemoryRegionInfo->regions[0].region.memory_class = I915_MEMORY_CLASS_SYSTEM;
45 queryMemoryRegionInfo->regions[0].region.memory_instance = 1;
46 queryMemoryRegionInfo->regions[0].probed_size = 2 * MemoryConstants::gigaByte;
47 queryMemoryRegionInfo->regions[1].region.memory_class = I915_MEMORY_CLASS_DEVICE;
48 queryMemoryRegionInfo->regions[1].region.memory_instance = 1;
49 queryMemoryRegionInfo->regions[1].probed_size = 2 * MemoryConstants::gigaByte;
50 }
51 }

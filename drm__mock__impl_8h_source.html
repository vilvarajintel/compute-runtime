44 virtual int handleRemainingRequests(unsigned long request, void *arg) override {
45 if ((request == DRM_IOCTL_I915_QUERY) && (arg != nullptr)) {
46 if (i915QuerySuccessCount == 0) {
47 return EINVAL;
48 }
49 i915QuerySuccessCount--;
50 auto query = static_cast<drm_i915_query *>(arg);
51 if (query->items_ptr == 0) {
52 return EINVAL;
53 }
54 for (auto i = 0u; i < query->num_items; i++) {
55 handleQueryItem(reinterpret_cast<drm_i915_query_item *>(query->items_ptr) + i);
56 }
57 return 0;
58 } else if (request == DRM_IOCTL_I915_GEM_MMAP_OFFSET) {
59 auto mmap_arg = static_cast<drm_i915_gem_mmap_offset *>(arg);
60 mmapOffsetFlagsReceived = mmap_arg->flags;
61 mmap_arg->offset = offset;
62 return mmapOffsetRetVal;
63 }
64 return handleKernelSpecificRequests(request, arg);
65 }

67 virtual void handleQueryItem(drm_i915_query_item *queryItem) {
68 switch (queryItem->query_id) {
69 case DRM_I915_QUERY_MEMORY_REGIONS:
70 if (queryMemoryRegionInfoSuccessCount == 0) {
71 queryItem->length = -EINVAL;
72 } else {
73 queryMemoryRegionInfoSuccessCount--;
74 auto numberOfLocalMemories = 1u;
75 auto numberOfRegions = 1u + numberOfLocalMemories;
76 int regionInfoSize = sizeof(drm_i915_query_memory_regions) + numberOfRegions * sizeof(drm_i915_memory_region_info);
77 if (queryItem->length == 0) {
78 queryItem->length = regionInfoSize;
79 } else {
80 EXPECT_EQ(regionInfoSize, queryItem->length);
81 auto queryMemoryRegionInfo = reinterpret_cast<drm_i915_query_memory_regions *>(queryItem->data_ptr);
82 EXPECT_EQ(0u, queryMemoryRegionInfo->num_regions);
83 queryMemoryRegionInfo->num_regions = numberOfRegions;
84 queryMemoryRegionInfo->regions[0].region.memory_class = I915_MEMORY_CLASS_SYSTEM;
85 queryMemoryRegionInfo->regions[0].region.memory_instance = 1;
86 queryMemoryRegionInfo->regions[0].probed_size = 2 * MemoryConstants::gigaByte;
87 queryMemoryRegionInfo->regions[1].region.memory_class = I915_MEMORY_CLASS_DEVICE;
88 queryMemoryRegionInfo->regions[1].region.memory_instance = 1;
89 queryMemoryRegionInfo->regions[1].probed_size = 2 * MemoryConstants::gigaByte;
90 }
91 }
92 break;
93 }

96 virtual int handleKernelSpecificRequests(unsigned long request, void *arg) {
97 if (request == DRM_IOCTL_I915_GEM_CREATE_EXT) {
98 auto createExtParams = static_cast<drm_i915_gem_create_ext *>(arg);
99 if (createExtParams->size == 0) {
100 return EINVAL;
101 }
102 createExtParams->handle = 1u;
103 this->createExt = *createExtParams;
104 auto extMemRegions = reinterpret_cast<drm_i915_gem_create_ext_memory_regions *>(createExt.extensions);
105 if (extMemRegions->base.name != I915_GEM_CREATE_EXT_MEMORY_REGIONS) {
106 return EINVAL;
107 }
108 this->numRegions = extMemRegions->num_regions;
109 this->memRegions = *reinterpret_cast<drm_i915_gem_memory_class_instance *>(extMemRegions->regions);
110 if (this->numRegions == 0) {
111 return EINVAL;
112 }
113 if ((this->memRegions.memory_class != I915_MEMORY_CLASS_SYSTEM) && (this->memRegions.memory_class != I915_MEMORY_CLASS_DEVICE)) {
114 return EINVAL;
115 }
116 return gemCreateExtRetVal;
117 }
118 return -1;

153#define ZE_HANDLE_TRACER_RECURSION(ze_api_ptr, ...) \
154 do { \
155 if (L0::tracingInProgress) { \
156 return ze_api_ptr(__VA_ARGS__); \
157 } \
158 L0::tracingInProgress = 1; \
159 } while (0)

161#define ZE_GEN_TRACER_ARRAY_ENTRY(callbackPtr, tracerArray, tracerArrayIndex, callbackType, callbackCategory, callbackFunction) \
162 do { \
163 callbackPtr = tracerArray->tracerArrayEntries[tracerArrayIndex].callbackType.callbackCategory.callbackFunction; \
164 } while (0)

166#define ZE_GEN_PER_API_CALLBACK_STATE(perApiCallbackData, tracerType, callbackCategory, callbackFunctionType) \
167 L0::tracer_array_t *currentTracerArray; \
168 currentTracerArray = (L0::tracer_array_t *)L0::pGlobalAPITracerContextImp->getActiveTracersList(); \
169 if (currentTracerArray) { \
170 for (size_t i = 0; i < currentTracerArray->tracerArrayCount; i++) { \
171 tracerType prologueCallbackPtr; \
172 tracerType epilogue_callback_ptr; \
173 ZE_GEN_TRACER_ARRAY_ENTRY(prologueCallbackPtr, currentTracerArray, i, corePrologues, callbackCategory, callbackFunctionType); \
174 ZE_GEN_TRACER_ARRAY_ENTRY(epilogue_callback_ptr, currentTracerArray, i, coreEpilogues, callbackCategory, callbackFunctionType); \
175 \
176 L0::APITracerCallbackStateImp<tracerType> prologCallback; \
177 prologCallback.current_api_callback = prologueCallbackPtr; \
178 prologCallback.pUserData = currentTracerArray->tracerArrayEntries[i].pUserData; \
179 perApiCallbackData.prologCallbacks.push_back(prologCallback); \
180 \
181 L0::APITracerCallbackStateImp<tracerType> epilogCallback; \
182 epilogCallback.current_api_callback = epilogue_callback_ptr; \
183 epilogCallback.pUserData = currentTracerArray->tracerArrayEntries[i].pUserData; \
184 perApiCallbackData.epilogCallbacks.push_back(epilogCallback); \
185 } \
186 }

188template <typename TFunction_pointer, typename TParams, typename TTracer, typename TTracerPrologCallbacks, typename TTracerEpilogCallbacks, typename... Args>
189ze_result_t APITracerWrapperImp(TFunction_pointer zeApiPtr,
190 TParams paramsStruct,
191 TTracer apiOrdinal,
192 TTracerPrologCallbacks prologCallbacks,
193 TTracerEpilogCallbacks epilogCallbacks,
194 Args &&...args) {
195 ze_result_t ret = ZE_RESULT_SUCCESS;
196 std::vector<APITracerCallbackStateImp<TTracer>> *callbacks_prologs = &prologCallbacks;
197
198 std::vector<void *> ppTracerInstanceUserData;
199 ppTracerInstanceUserData.resize(callbacks_prologs->size());
200
201 for (size_t i = 0; i < callbacks_prologs->size(); i++) {
202 if (callbacks_prologs->at(i).current_api_callback != nullptr)
203 callbacks_prologs->at(i).current_api_callback(paramsStruct, ret, callbacks_prologs->at(i).pUserData, &ppTracerInstanceUserData[i]);
204 }
205 ret = zeApiPtr(args...);
206 std::vector<APITracerCallbackStateImp<TTracer>> *callbacksEpilogs = &epilogCallbacks;
207 for (size_t i = 0; i < callbacksEpilogs->size(); i++) {
208 if (callbacksEpilogs->at(i).current_api_callback != nullptr)
209 callbacksEpilogs->at(i).current_api_callback(paramsStruct, ret, callbacksEpilogs->at(i).pUserData, &ppTracerInstanceUserData[i]);
210 }
211 L0::tracingInProgress = 0;
212 L0::pGlobalAPITracerContextImp->releaseActivetracersList();
213 return ret;
214}

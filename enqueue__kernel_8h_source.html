23template <typename GfxFamily>
24cl_int CommandQueueHw<GfxFamily>::enqueueKernel(
25 Kernel *pKernel,
26 cl_uint workDim,
27 const size_t *globalWorkOffsetIn,
28 const size_t *globalWorkSizeIn,
29 const size_t *localWorkSizeIn,
30 cl_uint numEventsInWaitList,
31 const cl_event *eventWaitList,
32 cl_event *event) {
33
34 if (workDim > device->getDeviceInfo().maxWorkItemDimensions) {
35 return CL_INVALID_WORK_DIMENSION;
36 }
37
38 size_t region[3] = {1, 1, 1};
39 size_t globalWorkOffset[3] = {0, 0, 0};
40 size_t workGroupSize[3] = {1, 1, 1};
41 size_t enqueuedLocalWorkSize[3] = {0, 0, 0};
42
43 auto &kernel = *pKernel;
44 const auto &kernelInfo = kernel.getKernelInfo();
45
46 if (!kernel.isPatched()) {
47 if (event) {
48 *event = nullptr;
49 }
50
51 return CL_INVALID_KERNEL_ARGS;
52 }
53
54 if (kernel.isUsingSharedObjArgs()) {
55 kernel.resetSharedObjectsPatchAddresses();
56 }
57
58 bool haveRequiredWorkGroupSize = false;
59
60 if (kernelInfo.kernelDescriptor.kernelAttributes.requiredWorkgroupSize[0] != 0) {
61 haveRequiredWorkGroupSize = true;
62 }
63
64 size_t remainder = 0;
65 size_t totalWorkItems = 1u;
66 const size_t *localWkgSizeToPass = localWorkSizeIn ? workGroupSize : nullptr;
67 size_t reqdWorkgroupSize[3] = {};
68
69 for (auto i = 0u; i < workDim; i++) {
70 region[i] = globalWorkSizeIn ? globalWorkSizeIn[i] : 0;
71 globalWorkOffset[i] = globalWorkOffsetIn
72 ? globalWorkOffsetIn[i]
73 : 0;
74
75 if (localWorkSizeIn) {
76 if (haveRequiredWorkGroupSize) {
77 if (kernelInfo.kernelDescriptor.kernelAttributes.requiredWorkgroupSize[i] != localWorkSizeIn[i]) {
78 return CL_INVALID_WORK_GROUP_SIZE;
79 }
80 }
81 if (localWorkSizeIn[i] == 0) {
82 return CL_INVALID_WORK_GROUP_SIZE;
83 }
84 if (kernel.getAllowNonUniform()) {
85 workGroupSize[i] = std::min(localWorkSizeIn[i], std::max(static_cast<size_t>(1), globalWorkSizeIn[i]));
86 } else {
87 workGroupSize[i] = localWorkSizeIn[i];
88 }
89 enqueuedLocalWorkSize[i] = localWorkSizeIn[i];
90 totalWorkItems *= localWorkSizeIn[i];
91 }
92
93 remainder += region[i] % workGroupSize[i];
94 }
95
96 if (remainder != 0 && !kernel.getAllowNonUniform()) {
97 return CL_INVALID_WORK_GROUP_SIZE;
98 }
99
100 if (haveRequiredWorkGroupSize) {
101 reqdWorkgroupSize[0] = kernelInfo.kernelDescriptor.kernelAttributes.requiredWorkgroupSize[0];
102 reqdWorkgroupSize[1] = kernelInfo.kernelDescriptor.kernelAttributes.requiredWorkgroupSize[1];
103 reqdWorkgroupSize[2] = kernelInfo.kernelDescriptor.kernelAttributes.requiredWorkgroupSize[2];
104 localWkgSizeToPass = reqdWorkgroupSize;
105 }
106
107 NullSurface s;
108 Surface *surfaces[] = {&s};
109
110 if (context->isProvidingPerformanceHints()) {
111 if (kernel.hasPrintfOutput()) {
112 context->providePerformanceHint(CL_CONTEXT_DIAGNOSTICS_LEVEL_NEUTRAL_INTEL, PRINTF_DETECTED_IN_KERNEL, kernelInfo.kernelDescriptor.kernelMetadata.kernelName.c_str());
113 }
114 if (kernel.requiresCoherency()) {
115 context->providePerformanceHint(CL_CONTEXT_DIAGNOSTICS_LEVEL_NEUTRAL_INTEL, KERNEL_REQUIRES_COHERENCY, kernelInfo.kernelDescriptor.kernelMetadata.kernelName.c_str());
116 }
117 }
118
119 if (kernelInfo.builtinDispatchBuilder != nullptr) {
120 cl_int err = kernelInfo.builtinDispatchBuilder->validateDispatch(&kernel, workDim, Vec3<size_t>(region), Vec3<size_t>(workGroupSize), Vec3<size_t>(globalWorkOffset));
121 if (err != CL_SUCCESS)
122 return err;
123 }
124
125 DBG_LOG(PrintDispatchParameters, "Kernel: ", kernelInfo.kernelDescriptor.kernelMetadata.kernelName,
126 ",LWS:, ", localWorkSizeIn ? localWorkSizeIn[0] : 0,
127 ",", localWorkSizeIn ? localWorkSizeIn[1] : 0,
128 ",", localWorkSizeIn ? localWorkSizeIn[2] : 0,
129 ",GWS:,", globalWorkSizeIn[0],
130 ",", globalWorkSizeIn[1],
131 ",", globalWorkSizeIn[2],
132 ",SIMD:, ", kernelInfo.getMaxSimdSize());
133
134 if (totalWorkItems > kernel.getMaxKernelWorkGroupSize()) {
135 return CL_INVALID_WORK_GROUP_SIZE;
136 }
137
138 enqueueHandler<CL_COMMAND_NDRANGE_KERNEL>(
139 surfaces,
140 false,
141 &kernel,
142 workDim,
143 globalWorkOffset,
144 region,
145 localWkgSizeToPass,
146 enqueuedLocalWorkSize,
147 numEventsInWaitList,
148 eventWaitList,
149 event);
150
151 return CL_SUCCESS;
152}

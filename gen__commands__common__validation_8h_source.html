20template <typename FamilyType>
21void validateStateBaseAddress(uint64_t indirectObjectHeapBase,
22 uint64_t instructionHeapBaseAddress,
23 IndirectHeap *pDSH,
24 IndirectHeap *pIOH,
25 IndirectHeap *pSSH,
26 GenCmdList::iterator &startCommand,
27 GenCmdList::iterator &endCommand,
28 GenCmdList &cmdList,
29 uint64_t expectedGeneralStateHeapBaseAddress) {
30 typedef typename FamilyType::STATE_BASE_ADDRESS STATE_BASE_ADDRESS;
31 // All state should be programmed before walker
32 auto itorCmd = find<STATE_BASE_ADDRESS *>(startCommand, endCommand);
33 ASSERT_NE(endCommand, itorCmd);
34
35 auto *cmd = (STATE_BASE_ADDRESS *)*itorCmd;
36
37 // Verify all addresses are getting programmed
38 EXPECT_TRUE(cmd->getDynamicStateBaseAddressModifyEnable());
39 EXPECT_TRUE(cmd->getGeneralStateBaseAddressModifyEnable());
40 EXPECT_TRUE(cmd->getSurfaceStateBaseAddressModifyEnable());
41 EXPECT_TRUE(cmd->getIndirectObjectBaseAddressModifyEnable());
42 EXPECT_TRUE(cmd->getInstructionBaseAddressModifyEnable());
43
44 EXPECT_EQ(pDSH->getGraphicsAllocation()->getGpuAddress(), cmd->getDynamicStateBaseAddress());
45 // Stateless accesses require GSH.base to be 0.
46 EXPECT_EQ(expectedGeneralStateHeapBaseAddress, cmd->getGeneralStateBaseAddress());
47 EXPECT_EQ(pSSH->getGraphicsAllocation()->getGpuAddress(), cmd->getSurfaceStateBaseAddress());
48 EXPECT_EQ(pIOH->getGraphicsAllocation()->getGpuBaseAddress(), cmd->getIndirectObjectBaseAddress());
49 EXPECT_EQ(instructionHeapBaseAddress, cmd->getInstructionBaseAddress());
50
51 // Verify all sizes are getting programmed
52 EXPECT_TRUE(cmd->getDynamicStateBufferSizeModifyEnable());
53 EXPECT_TRUE(cmd->getGeneralStateBufferSizeModifyEnable());
54 EXPECT_TRUE(cmd->getIndirectObjectBufferSizeModifyEnable());
55 EXPECT_TRUE(cmd->getInstructionBufferSizeModifyEnable());
56
57 EXPECT_EQ(pDSH->getMaxAvailableSpace(), cmd->getDynamicStateBufferSize() * MemoryConstants::pageSize);
58 EXPECT_NE(0u, cmd->getGeneralStateBufferSize());
59 EXPECT_EQ(MemoryConstants::sizeOf4GBinPageEntities, cmd->getIndirectObjectBufferSize());
60 EXPECT_EQ(MemoryConstants::sizeOf4GBinPageEntities, cmd->getInstructionBufferSize());
61
62 // Generically validate this command
63 FamilyType::PARSE::template validateCommand<STATE_BASE_ADDRESS *>(cmdList.begin(), itorCmd);
64}

66template <typename FamilyType>
67void validateL3Programming(GenCmdList &cmdList, GenCmdList::iterator &itorWalker) {
68 typedef typename FamilyType::PARSE PARSE;
69 typedef typename PARSE::MI_LOAD_REGISTER_IMM MI_LOAD_REGISTER_IMM;
70
71 auto itorCmd = findMmio<FamilyType>(cmdList.begin(), itorWalker, L3CNTLRegisterOffset<FamilyType>::registerOffset);
72 if (UnitTestHelper<FamilyType>::isL3ConfigProgrammable()) {
73 // All state should be programmed before walker
74 ASSERT_NE(itorWalker, itorCmd);
75
76 auto *cmd = genCmdCast<MI_LOAD_REGISTER_IMM *>(*itorCmd);
77 ASSERT_NE(nullptr, cmd);
78
79 auto registerOffset = L3CNTLRegisterOffset<FamilyType>::registerOffset;
80 EXPECT_EQ(registerOffset, cmd->getRegisterOffset());
81 auto l3Cntlreg = cmd->getDataDword();
82 auto numURBWays = (l3Cntlreg >> 1) & 0x7f;
83 auto L3ClientPool = (l3Cntlreg >> 25) & 0x7f;
84 EXPECT_NE(0u, numURBWays);
85 EXPECT_NE(0u, L3ClientPool);
86 } else {
87 ASSERT_EQ(itorWalker, itorCmd);
88 }
89}

93 typedef typename FamilyType::MEDIA_VFE_STATE MEDIA_VFE_STATE;

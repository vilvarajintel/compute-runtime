33 void setBufferSurfaceState(uint32_t argIndex, void *address, NEO::GraphicsAllocation *alloc) override {
34 uint64_t baseAddress = alloc->getGpuAddressToPatch();
35 auto sshAlignmentMask = NEO::EncodeSurfaceState<GfxFamily>::getSurfaceBaseAddressAlignmentMask();
36
37 // Remove misalligned bytes, accounted for in in bufferOffset patch token
38 baseAddress &= sshAlignmentMask;
39 auto misalignedSize = ptrDiff(alloc->getGpuAddressToPatch(), baseAddress);
40 auto offset = ptrDiff(address, reinterpret_cast<void *>(baseAddress));
41 size_t bufferSizeForSsh = alloc->getUnderlyingBufferSize();
42 auto argInfo = kernelImmData->getDescriptor().payloadMappings.explicitArgs[argIndex].as<NEO::ArgDescPointer>();
43 bool offsetWasPatched = NEO::patchNonPointer<uint32_t, uint32_t>(ArrayRef<uint8_t>(this->crossThreadData.get(), this->crossThreadDataSize),
44 argInfo.bufferOffset, static_cast<uint32_t>(offset));
45 if (false == offsetWasPatched) {
46 // fallback to handling offset in surface state
47 baseAddress = reinterpret_cast<uintptr_t>(address);
48 bufferSizeForSsh -= offset;
49 DEBUG_BREAK_IF(baseAddress != (baseAddress & sshAlignmentMask));
50 offset = 0;
51 }
52 void *surfaceStateAddress = nullptr;
53 auto surfaceState = GfxFamily::cmdInitRenderSurfaceState;
54 if (NEO::isValidOffset(argInfo.bindless)) {
55 surfaceStateAddress = patchBindlessSurfaceState(alloc, argInfo.bindless);
56 } else {
57 surfaceStateAddress = ptrOffset(surfaceStateHeapData.get(), argInfo.bindful);
58 surfaceState = *reinterpret_cast<typename GfxFamily::RENDER_SURFACE_STATE *>(surfaceStateAddress);
59 }
60 uint64_t bufferAddressForSsh = baseAddress;
61 auto alignment = NEO::EncodeSurfaceState<GfxFamily>::getSurfaceBaseAddressAlignment();
62 bufferSizeForSsh += misalignedSize;
63 bufferSizeForSsh = alignUp(bufferSizeForSsh, alignment);
64
65 bool l3Enabled = true;
66 // Allocation MUST be cacheline (64 byte) aligned in order to enable L3 caching otherwise Heap corruption will occur coming from the KMD.
67 // Most commonly this issue will occur with Host Point Allocations from customers.
68 l3Enabled = isL3Capable(*alloc);
69
70 Device *device = module->getDevice();
71 NEO::Device *neoDevice = device->getNEODevice();
72
73 auto allocData = device->getDriverHandle()->getSvmAllocsManager()->getSVMAlloc(reinterpret_cast<void *>(alloc->getGpuAddress()));
74 if (allocData && allocData->allocationFlagsProperty.flags.locallyUncachedResource) {
75 l3Enabled = false;
76 }
77
78 if (l3Enabled == false) {
79 this->kernelRequiresQueueUncachedMocsCount++;
80 }
81
82 NEO::EncodeSurfaceStateArgs args;
83 args.outMemory = &surfaceState;
84 args.graphicsAddress = bufferAddressForSsh;
85 args.size = bufferSizeForSsh;
86 args.mocs = device->getMOCS(l3Enabled, false);
87 args.numAvailableDevices = neoDevice->getNumGenericSubDevices();
88 args.allocation = alloc;
89 args.gmmHelper = neoDevice->getGmmHelper();
90 args.useGlobalAtomics = kernelImmData->getDescriptor().kernelAttributes.flags.useGlobalAtomics;
91 args.areMultipleSubDevicesInContext = args.numAvailableDevices > 1;
92 args.implicitScaling = device->isImplicitScalingCapable();
93
94 NEO::EncodeSurfaceState<GfxFamily>::encodeBuffer(args);
95 *reinterpret_cast<typename GfxFamily::RENDER_SURFACE_STATE *>(surfaceStateAddress) = surfaceState;
96 }

98 void evaluateIfRequiresGenerationOfLocalIdsByRuntime(const NEO::KernelDescriptor &kernelDescriptor) override {
99 size_t localWorkSizes[3];
100 localWorkSizes[0] = this->groupSize[0];
101 localWorkSizes[1] = this->groupSize[1];
102 localWorkSizes[2] = this->groupSize[2];
103
104 kernelRequiresGenerationOfLocalIdsByRuntime = NEO::EncodeDispatchKernel<GfxFamily>::isRuntimeLocalIdsGenerationRequired(
105 kernelDescriptor.kernelAttributes.numLocalIdChannels,
106 localWorkSizes,
107 std::array<uint8_t, 3>{
108 {kernelDescriptor.kernelAttributes.workgroupWalkOrder[0],
109 kernelDescriptor.kernelAttributes.workgroupWalkOrder[1],
110 kernelDescriptor.kernelAttributes.workgroupWalkOrder[2]}},
111 kernelDescriptor.kernelAttributes.flags.requiresWorkgroupWalkOrder,
112 requiredWorkgroupOrder,
113 kernelDescriptor.kernelAttributes.simdSize);
114 }

75template <typename Param>
76class ThkWrapper {
77 typedef NTSTATUS(APIENTRY *Func)(Param);
78
79 public:
80 Func mFunc = nullptr;
81
82 inline NTSTATUS operator()(Param param) const {
83 if (KMD_PROFILING) {
84 SYSTEM_ENTER()
85 NTSTATUS Status;
86 Status = mFunc(param);
87 SYSTEM_LEAVE(getId<Param>());
88 return Status;
89 } else {
90 return mFunc(param);
91 }
92 }
93
94 ThkWrapper &operator=(void *func) {
95 mFunc = reinterpret_cast<decltype(mFunc)>(func);
96 return *this;
97 }
98
99 ThkWrapper &operator=(Func func) {
100 mFunc = func;
101 return *this;
102 }
103
104 // This operator overload is for implicit casting ThkWrapper struct to Function Pointer in GetPfn methods like GetEscapePfn() or for comparing against NULL function pointer
105 operator Func() const {
106 return mFunc;
107 }
108};

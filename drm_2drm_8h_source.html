1105 *
1106 * \sa drmCommandNone(), drmCommandRead(), drmCommandWrite(), and

1112/*
1113 * Header for events written back to userspace on the drm fd. The
1114 * type defines the type of event, the length specifies the total
1115 * length of the event (including the header), and user_data is
1116 * typically a 64 bit value passed with the ioctl that triggered the
1117 * event. A read on the drm fd will always only return complete
1118 * events, that is, if for example the read buffer is 100 bytes, and
1119 * there are two 64 byte events pending, only one will be returned.
1120 *
1121 * Event types 0 - 0x7fffffff are generic drm events, 0x80000000 and
1122 * up are chipset specific.
1123 */
1124struct drm_event {
1125 __u32 type;
1126 __u32 length;
1127};

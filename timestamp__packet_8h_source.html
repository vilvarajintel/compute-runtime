31template <typename TSize>
32class TimestampPackets : public TagTypeBase {
33 protected:
34 struct Packet {
35 TSize contextStart = 1u;
36 TSize globalStart = 1u;
37 TSize contextEnd = 1u;
38 TSize globalEnd = 1u;
39 };
40
41 public:
42 static constexpr AllocationType getAllocationType() {
43 return AllocationType::TIMESTAMP_PACKET_TAG_BUFFER;
44 }
45
46 static constexpr TagNodeType getTagNodeType() { return TagNodeType::TimestampPacket; }
47
48 static constexpr size_t getSinglePacketSize() { return sizeof(Packet); }
49
50 void initialize() {
51 for (auto &packet : packets) {
52 packet.contextStart = 1u;
53 packet.globalStart = 1u;
54 packet.contextEnd = 1u;
55 packet.globalEnd = 1u;
56 }
57 }
58
59 void assignDataToAllTimestamps(uint32_t packetIndex, void *source) {
60 memcpy_s(&packets[packetIndex], sizeof(Packet), source, sizeof(Packet));
61 }
62
63 static constexpr size_t getGlobalStartOffset() { return offsetof(Packet, globalStart); }
64 static constexpr size_t getContextStartOffset() { return offsetof(Packet, contextStart); }
65 static constexpr size_t getContextEndOffset() { return offsetof(Packet, contextEnd); }
66 static constexpr size_t getGlobalEndOffset() { return offsetof(Packet, globalEnd); }
67
68 uint64_t getContextStartValue(uint32_t packetIndex) const { return static_cast<uint64_t>(packets[packetIndex].contextStart); }
69 uint64_t getGlobalStartValue(uint32_t packetIndex) const { return static_cast<uint64_t>(packets[packetIndex].globalStart); }
70 uint64_t getContextEndValue(uint32_t packetIndex) const { return static_cast<uint64_t>(packets[packetIndex].contextEnd); }
71 uint64_t getGlobalEndValue(uint32_t packetIndex) const { return static_cast<uint64_t>(packets[packetIndex].globalEnd); }
72
73 void const *getContextEndAddress(uint32_t packetIndex) const { return static_cast<void const *>(&packets[packetIndex].contextEnd); }
74
75 protected:
76 Packet packets[TimestampPacketSizeControl::preferredPacketCount];
77};

125 template <typename GfxFamily>
126 static void programSemaphore(LinearStream &cmdStream, TagNodeBase &timestampPacketNode) {
127 using COMPARE_OPERATION = typename GfxFamily::MI_SEMAPHORE_WAIT::COMPARE_OPERATION;
128 using MI_SEMAPHORE_WAIT = typename GfxFamily::MI_SEMAPHORE_WAIT;
129
130 auto compareAddress = getContextEndGpuAddress(timestampPacketNode);
131
132 for (uint32_t packetId = 0; packetId < timestampPacketNode.getPacketsUsed(); packetId++) {
133 uint64_t compareOffset = packetId * timestampPacketNode.getSinglePacketSize();
134 EncodeSempahore<GfxFamily>::addMiSemaphoreWaitCommand(cmdStream, compareAddress + compareOffset, 1, COMPARE_OPERATION::COMPARE_OPERATION_SAD_NOT_EQUAL_SDD);
135 }
136 }

162 template <typename GfxFamily, AuxTranslationDirection auxTranslationDirection>
163 static void programSemaphoreForAuxTranslation(LinearStream &cmdStream,
164 const TimestampPacketDependencies *timestampPacketDependencies,
165 const HardwareInfo &hwInfo) {
166 auto &container = (auxTranslationDirection == AuxTranslationDirection::AuxToNonAux)
167 ? timestampPacketDependencies->auxToNonAuxNodes
168 : timestampPacketDependencies->nonAuxToAuxNodes;
169
170 // cache flush after NDR, before NonAuxToAux
171 if (auxTranslationDirection == AuxTranslationDirection::NonAuxToAux && timestampPacketDependencies->cacheFlushNodes.peekNodes().size() > 0) {
172 UNRECOVERABLE_IF(timestampPacketDependencies->cacheFlushNodes.peekNodes().size() != 1);
173 auto cacheFlushTimestampPacketGpuAddress = getContextEndGpuAddress(*timestampPacketDependencies->cacheFlushNodes.peekNodes()[0]);
174
175 PipeControlArgs args;
176 args.dcFlushEnable = MemorySynchronizationCommands<GfxFamily>::getDcFlushEnable(true, hwInfo);
177 MemorySynchronizationCommands<GfxFamily>::addPipeControlAndProgramPostSyncOperation(
178 cmdStream, GfxFamily::PIPE_CONTROL::POST_SYNC_OPERATION::POST_SYNC_OPERATION_WRITE_IMMEDIATE_DATA,
179 cacheFlushTimestampPacketGpuAddress, 0, hwInfo, args);
180 }
181
182 for (auto &node : container.peekNodes()) {
183 TimestampPacketHelper::programSemaphore<GfxFamily>(cmdStream, *node);
184 }
185 }

188 static size_t getRequiredCmdStreamSizeForAuxTranslationNodeDependency(size_t count, const HardwareInfo &hwInfo, bool cacheFlushForBcsRequired) {
189 size_t size = count * TimestampPacketHelper::getRequiredCmdStreamSizeForNodeDependencyWithBlitEnqueue<GfxFamily>();

191 if (auxTranslationDirection == AuxTranslationDirection::NonAuxToAux && cacheFlushForBcsRequired) {
192 size += MemorySynchronizationCommands<GfxFamily>::getSizeForPipeControlWithPostSyncOperation(hwInfo);
193 }

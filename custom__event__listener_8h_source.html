99 void OnTestProgramEnd(const ::testing::UnitTest &unitTest) override {
100 int testsRun = unitTest.test_to_run_count();
101 int testsPassed = unitTest.successful_test_count();
102 int testsSkipped = unitTest.skipped_test_count();
103 int testsFailed = unitTest.failed_test_count();
104 int testsDisabled = unitTest.disabled_test_count();
105 auto timeElapsed = static_cast<int>(unitTest.elapsed_time());
106 std::string ultStatus = "PASSED";
107 std::string paddingS = "";
108 std::string paddingE = "";
109 if (unitTest.Failed()) {
110 ultStatus = "FAILED";
111 }
112 auto executionNameLen = strlen(NEO::executionName);
113
114 if (hardwarePrefix != "---") {
115 paddingS = std::string(hardwarePrefix.length() + executionNameLen, ' ');
116 paddingE = std::string(hardwarePrefix.length() + executionNameLen, '=');
117
118 fprintf(
119 stdout,
120 "\n"
121 "%s==================\n"
122 "== %s %ss %s ==\n"
123 "%s==================\n",
124 paddingE.c_str(), hardwarePrefix.c_str(), NEO::executionName, ultStatus.c_str(), paddingE.c_str());
125 } else {
126 paddingE = std::string(executionNameLen, '=');
127 fprintf(
128 stdout,
129 "\n"
130 "%s==================\n"
131 "== %ss %s ==\n"
132 "%s==================\n",
133 paddingE.c_str(), NEO::executionName, ultStatus.c_str(), paddingE.c_str());
134 }
135
136 fprintf(
137 stdout,
138 "Tests run: %d\n"
139 "Tests passed: %d\n"
140 "Tests skipped: %d\n"
141 "Tests failed: %d\n"
142 "Tests disabled: %d\n"
143 " Time elapsed: %d ms\n"
144 "%s==================\n",
145 testsRun,
146 testsPassed,
147 testsSkipped,
148 testsFailed,
149 testsDisabled,
150 timeElapsed,
151 paddingE.c_str());
152
153 for (auto failure : testFailures)
154 fprintf(
155 stdout,
156 "[ FAILED ][ %s ][ %u ] %s\n", hardwarePrefix.c_str(), failure.second, failure.first.c_str());
157 if (unitTest.Failed())
158 fprintf(
159 stdout,
160 "\n");
161
162 fflush(stdout);
163 }

168 int currentSeed = -1;

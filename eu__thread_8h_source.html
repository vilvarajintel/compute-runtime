68 bool stopThread(uint64_t memHandle) {
69 memoryHandle = memHandle;
70 if (state == State::Stopped) {
71 return false;
72 }
73 state = State::Stopped;
74 PRINT_DEBUGGER_THREAD_LOG("Stopped thread: %s", toString().c_str());
75 return true;
76 }

78 bool verifyStopped(uint8_t newCounter) {
79
80 PRINT_DEBUGGER_THREAD_LOG("EuThread::verifyStopped() Thread: %s newCounter == %d oldCounter == %d", toString().c_str(), (int32_t)newCounter, (int32_t)systemRoutineCounter);
81
82 if (newCounter == systemRoutineCounter) {
83 if (newCounter % 2 != 0) {
84 if (state == State::Running) {
85 PRINT_DEBUGGER_ERROR_LOG("Thread: %s state RUNNING when thread is stopped. Switching to STOPPED", toString().c_str());
86 DEBUG_BREAK_IF(true);
87 }
88 state = State::Stopped;
89 return true;
90 }
91 }
92
93 if (newCounter == (systemRoutineCounter + 2)) {
94 state = State::Stopped;
95 systemRoutineCounter = newCounter;
96 return true;
97 } else if (newCounter > systemRoutineCounter + 2) {
98 PRINT_DEBUGGER_ERROR_LOG("Thread: %s state out of sync.", toString().c_str());
99 DEBUG_BREAK_IF(true);
100 }
101
102 if (newCounter % 2 == 0) {
103 if (state == State::Stopped) {
104 PRINT_DEBUGGER_ERROR_LOG("Thread: %s state STOPPED when thread is running. Switching to RUNNING", toString().c_str());
105 DEBUG_BREAK_IF(true);
106 }
107 state = State::Running;
108 systemRoutineCounter = newCounter;
109
110 return false;
111 }
112
113 state = State::Stopped;
114 systemRoutineCounter = newCounter;
115 return true;
116 }

118 bool resumeThread() {
119 if (state != State::Stopped) {
120 PRINT_DEBUGGER_THREAD_LOG("Resuming already RUNNING thread: %s", toString().c_str());
121 return false;
122 }
123 PRINT_DEBUGGER_THREAD_LOG("Resumed thread: %s", toString().c_str());
124 state = State::Running;
125 memoryHandle = invalidHandle;
126 return true;
127 }

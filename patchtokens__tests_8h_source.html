38inline uint32_t pushBackString(const std::string &str, std::vector<uint8_t> &storage) {
39 auto offset = storage.size();
40 storage.insert(storage.end(), reinterpret_cast<const uint8_t *>(str.c_str()), reinterpret_cast<const uint8_t *>(str.c_str()) + str.size());
41 return static_cast<uint32_t>(offset);
42}

44inline uint32_t pushBackStringToken(const std::string &str, uint32_t stringIndex, std::vector<uint8_t> &outStream) {
45 auto off = outStream.size();
46 outStream.reserve(outStream.size() + sizeof(iOpenCL::SPatchString) + str.length());
47 outStream.resize(outStream.size() + sizeof(iOpenCL::SPatchString));
48 iOpenCL::SPatchString *tok = reinterpret_cast<iOpenCL::SPatchString *>(outStream.data() + off);
49 *tok = initToken<iOpenCL::SPatchString>(iOpenCL::PATCH_TOKEN::PATCH_TOKEN_STRING);
50 tok->StringSize = static_cast<uint32_t>(str.length());
51 tok->Size += tok->StringSize;
52 tok->Index = stringIndex;
53 pushBackString(str, outStream);
54 return static_cast<uint32_t>(off);
55};

57inline uint32_t pushBackArgInfoToken(std::vector<uint8_t> &outStream,
58 uint32_t argNum = 0,
59 const std::string &addressQualifier = "__global", const std::string &accessQualifier = "read_write",
60 const std::string &argName = "custom_arg", const std::string &typeName = "int*;", std::string typeQualifier = "const") {
61 auto off = outStream.size();
62 iOpenCL::SPatchKernelArgumentInfo tok = {};
63 tok.Token = iOpenCL::PATCH_TOKEN_KERNEL_ARGUMENT_INFO;
64 tok.AddressQualifierSize = static_cast<uint32_t>(addressQualifier.size());
65 tok.AccessQualifierSize = static_cast<uint32_t>(accessQualifier.size());
66 tok.ArgumentNameSize = static_cast<uint32_t>(argName.size());
67 tok.TypeNameSize = static_cast<uint32_t>(typeName.size());
68 tok.TypeQualifierSize = static_cast<uint32_t>(typeQualifier.size());
69 tok.Size = sizeof(iOpenCL::SPatchKernelArgumentInfo) + tok.AddressQualifierSize + tok.AccessQualifierSize + tok.ArgumentNameSize + tok.TypeNameSize + tok.TypeQualifierSize;
70
71 outStream.insert(outStream.end(), reinterpret_cast<const uint8_t *>(&tok), reinterpret_cast<const uint8_t *>(&tok) + sizeof(tok));
72 pushBackString(addressQualifier, outStream);
73 pushBackString(accessQualifier, outStream);
74 pushBackString(argName, outStream);
75 pushBackString(typeName, outStream);
76 pushBackString(typeQualifier, outStream);
77 return static_cast<uint32_t>(off);
78}

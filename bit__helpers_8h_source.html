15constexpr bool isBitSet(uint64_t field, uint64_t bitPosition) {
16 assert(bitPosition < std::numeric_limits<uint64_t>::digits); // undefined behavior
17 return (field & (1ull << bitPosition));
18}
19
20constexpr bool isAnyBitSet(uint64_t field, uint64_t checkedBits) {
21 return ((field & checkedBits) != 0);
22}
23
24constexpr bool isValueSet(uint64_t field, uint64_t value) {
25 assert(value != 0);
26 return ((field & value) == value);
27}
28
29constexpr bool isFieldValid(uint64_t field, uint64_t acceptedBits) {
30 return ((field & (~acceptedBits)) == 0);
31}
32
33constexpr uint64_t setBits(uint64_t field, bool newValue, uint64_t bitsToModify) {
34 if (newValue) {
35 return (field | bitsToModify);
36 }
37 return (field & (~bitsToModify));
38}

40constexpr uint64_t shiftLeftBy(uint64_t bitPosition) {
41 return (1ull << bitPosition);
42}

116inline SingleDeviceBinary unpackSingleDeviceBinary(const ArrayRef<const uint8_t> archive, const ConstStringRef requestedProductAbbreviation, const TargetDevice &requestedTargetDevice,
117 std::string &outErrReason, std::string &outWarning) {

120 if (isDeviceBinaryFormat<DeviceBinaryFormat::OclElf>(archive)) {

122 } else if (isDeviceBinaryFormat<DeviceBinaryFormat::Patchtokens>(archive)) {
123 return unpackSingleDeviceBinary<DeviceBinaryFormat::Patchtokens>(archive, requestedProductAbbreviation, requestedTargetDevice, outErrReason, outWarning);

171inline std::pair<DecodeError, DeviceBinaryFormat> decodeSingleDeviceBinary(ProgramInfo &dst, const SingleDeviceBinary &src, std::string &outErrReason, std::string &outWarning) {
172 std::pair<DecodeError, DeviceBinaryFormat> ret;
173 ret.first = DecodeError::InvalidBinary;
174 ret.second = DeviceBinaryFormat::Unknown;
175 if (isDeviceBinaryFormat<DeviceBinaryFormat::OclElf>(src.deviceBinary)) {
176 ret.second = DeviceBinaryFormat::OclElf;
177 ret.first = decodeSingleDeviceBinary<DeviceBinaryFormat::OclElf>(dst, src, outErrReason, outWarning);
178 } else if (isDeviceBinaryFormat<DeviceBinaryFormat::Patchtokens>(src.deviceBinary)) {
179 ret.second = DeviceBinaryFormat::Patchtokens;
180 ret.first = decodeSingleDeviceBinary<DeviceBinaryFormat::Patchtokens>(dst, src, outErrReason, outWarning);
181 } else if (isDeviceBinaryFormat<DeviceBinaryFormat::Archive>(src.deviceBinary)) {
182 ret.second = DeviceBinaryFormat::Archive;
183 ret.first = decodeSingleDeviceBinary<DeviceBinaryFormat::Archive>(dst, src, outErrReason, outWarning);
184 } else if (isDeviceBinaryFormat<DeviceBinaryFormat::Zebin>(src.deviceBinary)) {
185 ret.second = DeviceBinaryFormat::Zebin;
186 ret.first = decodeSingleDeviceBinary<DeviceBinaryFormat::Zebin>(dst, src, outErrReason, outWarning);
187 } else {
188 outErrReason = "Unknown format";
189 }
190 return ret;
191}

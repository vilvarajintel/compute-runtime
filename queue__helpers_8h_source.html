70inline cl_int getQueueInfo(CommandQueue *queue,
71 cl_command_queue_info paramName,
72 size_t paramValueSize,
73 void *paramValue,
74 size_t *paramValueSizeRet) {
75
76 cl_int retVal = CL_SUCCESS;
77 GetInfoHelper getInfoHelper(paramValue, paramValueSize, paramValueSizeRet);
78
79 switch (paramName) {
80 case CL_QUEUE_CONTEXT:
81 retVal = changeGetInfoStatusToCLResultType(getInfoHelper.set<cl_context>(queue->getContextPtr()));
82 break;
83 case CL_QUEUE_DEVICE: {
84 Device &device = queue->getDevice();
85 retVal = changeGetInfoStatusToCLResultType(getInfoHelper.set<cl_device_id>(device.getSpecializedDevice<ClDevice>()));
86 break;
87 }
88 case CL_QUEUE_REFERENCE_COUNT:
89 retVal = changeGetInfoStatusToCLResultType(getInfoHelper.set<cl_int>(queue->getReference()));
90 break;
91 case CL_QUEUE_PROPERTIES:
92 retVal = changeGetInfoStatusToCLResultType(getInfoHelper.set<cl_command_queue_properties>(queue->getCommandQueueProperties()));
93 break;
94 case CL_QUEUE_DEVICE_DEFAULT:
95 retVal = changeGetInfoStatusToCLResultType(getInfoHelper.set<cl_command_queue>(nullptr));
96 break;
97 case CL_QUEUE_SIZE:
98 retVal = CL_INVALID_COMMAND_QUEUE;
99 break;
100 case CL_QUEUE_PROPERTIES_ARRAY: {
101 auto &propertiesVector = queue->getPropertiesVector();
102 auto source = propertiesVector.data();
103 auto sourceSize = propertiesVector.size() * sizeof(cl_queue_properties);
104 auto getInfoStatus = GetInfo::getInfo(paramValue, paramValueSize, source, sourceSize);
105 retVal = changeGetInfoStatusToCLResultType(getInfoStatus);
106 GetInfo::setParamValueReturnSize(paramValueSizeRet, sourceSize, getInfoStatus);
107 break;
108 }
109 default:
110 getHostQueueInfo(queue, paramName, getInfoHelper, retVal);
111 break;
112 }
113
114 return retVal;
115}

130template <typename returnType>
131returnType getCmdQueueProperties(const cl_queue_properties *properties,
132 cl_queue_properties propertyName = CL_QUEUE_PROPERTIES,
133 bool *foundValue = nullptr) {
134 if (properties != nullptr) {
135 while (*properties != 0) {
136 if (*properties == propertyName) {
137 if (foundValue) {
138 *foundValue = true;
139 }
140 return static_cast<returnType>(*(properties + 1));
141 }
142 properties += 2;
143 }
144 }
145
146 if (foundValue) {
147 *foundValue = false;
148 }
149 return 0;
150}

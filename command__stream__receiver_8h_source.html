59enum class DispatchMode {

61 ImmediateDispatch, //everything is submitted to the HW immediately
62 AdaptiveDispatch, //dispatching is handled to async thread, which combines batch buffers basing on load (not implemented)

64 BatchedDispatch // dispatching is batched, explicit clFlush is required

17template <typename CT = int32_t>
18class RefCounter {
19 public:
20 RefCounter()
21 : val(0) {
22 }
23
24 CT peek() const {
25 CT curr = val.load();
26 DEBUG_BREAK_IF(curr < 0);
27 return curr;
28 }
29
30 void inc() {
31 [[maybe_unused]] CT curr = ++val;
32 DEBUG_BREAK_IF(curr < 1);
33 }
34
35 bool dec() {
36 CT curr = --val;
37 DEBUG_BREAK_IF(curr < 0);
38 return (curr == 0);
39 }
40
41 CT decAndReturnCurrent() {
42 return --val;
43 }
44
45 bool peekIsZero() const {
46 return (val == 0);
47 }
48
49 protected:
50 std::atomic<CT> val;
51};

56template <typename DataType>
57class unique_ptr_if_unused : public std::unique_ptr<DataType, void (*)(DataType *)> {
58 using DeleterFuncType = void (*)(DataType *);
59
60 public:
61 unique_ptr_if_unused()
62 : std::unique_ptr<DataType, DeleterFuncType>(nullptr, dontDelete) {
63 }
64
65 unique_ptr_if_unused(DataType *ptr, bool unused, DeleterFuncType customDeleter = nullptr)
66 : std::unique_ptr<DataType, DeleterFuncType>(ptr, unused ? chooseDeleter(ptr, customDeleter) : dontDelete) {
67 }
68
69 bool isUnused() const {
70 return (this->get_deleter() != dontDelete);
71 }
72
73 private:
74 static DeleterFuncType chooseDeleter(DataType *inPtr, DeleterFuncType customDeleter) {
75 DeleterFuncType deleter = customDeleter;
76 if (customDeleter == nullptr) {
77 deleter = getObjDeleter(inPtr);
78 }
79
80 if (deleter == nullptr) {
81 deleter = &doDelete;
82 }
83
84 return deleter;
85 }
86
87 template <typename DT = DataType>
88 static typename std::enable_if<std::is_base_of<ReferenceTrackedObject<DataType>, DT>::value, DeleterFuncType>::type getObjDeleter(DataType *inPtr) {
89 if (inPtr != nullptr) {
90 return inPtr->getCustomDeleter();
91 }
92 return nullptr;
93 }
94
95 template <typename DT = DataType>
96 static typename std::enable_if<!std::is_base_of<ReferenceTrackedObject<DataType>, DT>::value, DeleterFuncType>::type getObjDeleter(DataType *inPtr) {
97 return nullptr;
98 }
99
100 static void doDelete(DataType *ptr) {
101 delete ptr;
102 }
103
104 static void dontDelete(DataType *ptr) {
105 ;
106 }
107};

109// This class is needed for having both internal and external (api)
110// reference counters
111// Note : we need both counters because an OCL app can release an OCL object
112// while this object is still needed/valid (e.g. events with callbacks),
113// so we need to have a way of tracking internal usage of these object.
114// At the same time, we can't use one refcount for both internal and external
115// (retain/release api) usage because an OCL application can query object's
116// refcount (this refcount should not be contaminated by our internal usage models)
117// Note : internal refcount accumulates also api refcounts (i.e. incrementing/decrementing
118// api refcount will increment/decrement internal refcount as well) - so, object
119// deletion is based on single/atomic decision "if(--internalRefcount == 0)"
120template <typename DerivedClass>
121class ReferenceTrackedObject {
122 public:
123 virtual ~ReferenceTrackedObject();
124
125 int32_t getRefInternalCount() const {
126 return refInternal.peek();
127 }
128
129 void incRefInternal() {
130 refInternal.inc();
131 }
132
133 unique_ptr_if_unused<DerivedClass> decRefInternal() {
134 auto customDeleter = tryGetCustomDeleter();
135 auto current = refInternal.decAndReturnCurrent();
136 bool unused = (current == 0);
137 UNRECOVERABLE_IF(current < 0);
138 return unique_ptr_if_unused<DerivedClass>(static_cast<DerivedClass *>(this), unused, customDeleter);
139 }
140
141 int32_t getRefApiCount() const {
142 return refApi.peek();
143 }
144
145 void incRefApi() {
146 refApi.inc();
147 refInternal.inc();
148 }
149
150 unique_ptr_if_unused<DerivedClass> decRefApi() {
151 refApi.dec();
152 return decRefInternal();
153 }
154
155 using DeleterFuncType = void (*)(DerivedClass *);
156 DeleterFuncType getCustomDeleter() const {
157 return nullptr;
158 }
159
160 bool peekHasZeroRefcounts() const {
161 return refInternal.peekIsZero();
162 }
163
164 private:
165 DeleterFuncType tryGetCustomDeleter() const {
166 const DerivedClass *asDerivedObj = static_cast<const DerivedClass *>(this);
167 return asDerivedObj->getCustomDeleter();
168 }
169
170 RefCounter<> refInternal;
171 RefCounter<> refApi;
172};
173template <typename DerivedClass>
174inline ReferenceTrackedObject<DerivedClass>::~ReferenceTrackedObject() {
175 DEBUG_BREAK_IF(refInternal.peek() > 1);
176}

22template <typename GfxFamily>
23inline void setImageSurfaceState(typename GfxFamily::RENDER_SURFACE_STATE *surfaceState, const ImageInfo &imageInfo, Gmm *gmm, GmmHelper &gmmHelper, uint32_t cubeFaceIndex, uint64_t gpuAddress, const SurfaceOffsets &surfaceOffsets, bool isNV12Format) {
24 using RENDER_SURFACE_STATE = typename GfxFamily::RENDER_SURFACE_STATE;
25 using SURFACE_FORMAT = typename RENDER_SURFACE_STATE::SURFACE_FORMAT;
26 using AUXILIARY_SURFACE_MODE = typename RENDER_SURFACE_STATE::AUXILIARY_SURFACE_MODE;
27
28 auto imageCount = std::max(imageInfo.imgDesc.imageDepth, imageInfo.imgDesc.imageArraySize);
29 if (imageCount == 0) {
30 imageCount = 1;
31 }
32
33 bool isImageArray = imageInfo.imgDesc.imageArraySize > 1 &&
34 (imageInfo.imgDesc.imageType == ImageType::Image2DArray ||
35 imageInfo.imgDesc.imageType == ImageType::Image1DArray);
36
37 isImageArray |= (imageInfo.imgDesc.imageType == ImageType::Image2D || imageInfo.imgDesc.imageType == ImageType::Image2DArray) && DebugManager.flags.Force2dImageAsArray.get() == 1;
38
39 uint32_t renderTargetViewExtent = static_cast<uint32_t>(imageCount);
40 uint32_t minimumArrayElement = 0;
41 auto hAlign = RENDER_SURFACE_STATE::SURFACE_HORIZONTAL_ALIGNMENT_HALIGN_DEFAULT;
42 auto vAlign = RENDER_SURFACE_STATE::SURFACE_VERTICAL_ALIGNMENT_VALIGN_4;
43
44 if (gmm) {
45 hAlign = static_cast<typename RENDER_SURFACE_STATE::SURFACE_HORIZONTAL_ALIGNMENT>(gmm->gmmResourceInfo->getHAlignSurfaceState());
46 vAlign = static_cast<typename RENDER_SURFACE_STATE::SURFACE_VERTICAL_ALIGNMENT>(gmm->gmmResourceInfo->getVAlignSurfaceState());
47 }
48
49 if (cubeFaceIndex != __GMM_NO_CUBE_MAP) {
50 isImageArray = true;
51 renderTargetViewExtent = 1;
52 minimumArrayElement = cubeFaceIndex;
53 }
54
55 surfaceState->setAuxiliarySurfaceMode(AUXILIARY_SURFACE_MODE::AUXILIARY_SURFACE_MODE_AUX_NONE);
56 surfaceState->setAuxiliarySurfacePitch(1u);
57 surfaceState->setAuxiliarySurfaceQpitch(0u);
58 surfaceState->setAuxiliarySurfaceBaseAddress(0u);
59 surfaceState->setRenderTargetViewExtent(renderTargetViewExtent);
60 surfaceState->setMinimumArrayElement(minimumArrayElement);
61
62 // SurfaceQpitch is in rows but must be a multiple of VALIGN
63 surfaceState->setSurfaceQpitch(imageInfo.qPitch);
64
65 surfaceState->setSurfaceArray(isImageArray);
66
67 surfaceState->setSurfaceHorizontalAlignment(hAlign);
68 surfaceState->setSurfaceVerticalAlignment(vAlign);
69
70 uint32_t tileMode = gmm ? gmm->gmmResourceInfo->getTileModeSurfaceState()
71 : static_cast<uint32_t>(RENDER_SURFACE_STATE::TILE_MODE_LINEAR);
72
73 surfaceState->setTileMode(static_cast<typename RENDER_SURFACE_STATE::TILE_MODE>(tileMode));
74
75 surfaceState->setMemoryObjectControlState(gmmHelper.getMOCS(GMM_RESOURCE_USAGE_OCL_IMAGE));
76
77 surfaceState->setCoherencyType(RENDER_SURFACE_STATE::COHERENCY_TYPE_GPU_COHERENT);
78
79 surfaceState->setMultisampledSurfaceStorageFormat(RENDER_SURFACE_STATE::MULTISAMPLED_SURFACE_STORAGE_FORMAT::MULTISAMPLED_SURFACE_STORAGE_FORMAT_MSS);
80
81 surfaceState->setSurfaceBaseAddress(gpuAddress + surfaceOffsets.offset);
82 surfaceState->setXOffset(surfaceOffsets.xOffset);
83 surfaceState->setYOffset(surfaceOffsets.yOffset);
84
85 if (isNV12Format) {
86 surfaceState->setShaderChannelSelectAlpha(RENDER_SURFACE_STATE::SHADER_CHANNEL_SELECT_ONE);
87 surfaceState->setYOffsetForUOrUvPlane(surfaceOffsets.yOffsetForUVplane);
88 surfaceState->setXOffsetForUOrUvPlane(surfaceOffsets.xOffset);
89 } else {
90 surfaceState->setShaderChannelSelectAlpha(RENDER_SURFACE_STATE::SHADER_CHANNEL_SELECT_ALPHA);
91 surfaceState->setYOffsetForUOrUvPlane(0);
92 surfaceState->setXOffsetForUOrUvPlane(0);
93 }
94
95 surfaceState->setSurfaceFormat(static_cast<SURFACE_FORMAT>(imageInfo.surfaceFormat->GenxSurfaceFormat));
96 setFilterMode<GfxFamily>(surfaceState, gmmHelper.getHardwareInfo());
97}

99template <typename GfxFamily>
100inline void setImageSurfaceStateDimensions(typename GfxFamily::RENDER_SURFACE_STATE *surfaceState, const ImageInfo &imageInfo, uint32_t cubeFaceIndex, typename GfxFamily::RENDER_SURFACE_STATE::SURFACE_TYPE surfaceType) {
101 auto imageCount = std::max(imageInfo.imgDesc.imageDepth, imageInfo.imgDesc.imageArraySize);
102 if (imageCount == 0) {
103 imageCount = 1;
104 }
105
106 auto imageHeight = imageInfo.imgDesc.imageHeight;
107 if (imageHeight == 0) {
108 imageHeight = 1;
109 }
110
111 auto imageWidth = imageInfo.imgDesc.imageWidth;
112 if (imageWidth == 0) {
113 imageWidth = 1;
114 }
115
116 if (cubeFaceIndex != __GMM_NO_CUBE_MAP) {
117 imageCount = __GMM_MAX_CUBE_FACE - cubeFaceIndex;
118 }
119
120 surfaceState->setWidth(static_cast<uint32_t>(imageWidth));
121 surfaceState->setHeight(static_cast<uint32_t>(imageHeight));
122 surfaceState->setDepth(static_cast<uint32_t>(imageCount));
123 surfaceState->setSurfacePitch(static_cast<uint32_t>(imageInfo.imgDesc.imageRowPitch));
124 surfaceState->setSurfaceType(surfaceType);
125}

127template <typename GfxFamily>
128inline void setWidthForMediaBlockSurfaceState(typename GfxFamily::RENDER_SURFACE_STATE *surfaceState, const ImageInfo &imageInfo) {
129 auto elSize = imageInfo.surfaceFormat->ImageElementSizeInBytes;
130 auto numDwords = static_cast<uint32_t>(Math::divideAndRoundUp(imageInfo.imgDesc.imageWidth * elSize, sizeof(uint32_t)));
131 surfaceState->setWidth(numDwords);
132}

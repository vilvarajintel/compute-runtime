84template <typename ContainerT, typename T>
85inline void concatenateAppend(ContainerT &out, T &&arg) {
86 if ((false == out.empty()) && (*out.rbegin() != ' ')) {
87 out.push_back(' ');
88 }
89 out.insert(out.end(), data(arg), data(arg) + length(arg));
90}

92template <typename ContainerT, typename T, typename... RestT>
93inline void concatenateAppend(ContainerT &out, T &&arg, RestT &&...rest) {
94 concatenateAppend(out, std::forward<T>(arg));
95 concatenateAppend(out, std::forward<RestT>(rest)...);
96}

98template <typename T, typename... RestT>
99inline std::string concatenate(T &&arg, RestT &&...rest) {
100 std::string ret;
101 ret.reserve(nullterminateSize + concatenationLength(arg, rest...));
102 concatenateAppend(ret, std::forward<T>(arg), std::forward<RestT>(rest)...);
103 return ret;
104}

106template <size_t NumOptions>
107constexpr size_t concatenationLength(const ConstStringRef (&options)[NumOptions]) {
108 size_t ret = 0U;
109 for (auto opt : options) {
110 ret += spaceSeparatorSize + opt.length();
111 }
112 return (ret != 0U) ? ret - nullterminateSize : 0U;
113}

115template <typename ContainerT>
116inline bool extract(const ConstStringRef &toBeExtracted, ContainerT &options) {
117 const auto first{std::search(options.begin(), options.end(),
118 std::default_searcher{toBeExtracted.begin(), toBeExtracted.end()})};
119
120 if (first == options.end()) {
121 return false;
122 }
123
124 const auto last{std::next(first, toBeExtracted.length())};
125 options.erase(first, last);
126
127 return true;
128}

130template <size_t MaxLength = 256>
131class ConstConcatenation {
132 public:
133 template <size_t NumOptions>
134 constexpr ConstConcatenation(const ConstStringRef (&options)[NumOptions]) {
135 size_t i = 0U;
136 for (auto opt : options) {
137 for (size_t j = 0U, e = opt.length(); j < e; ++j, ++i) {
138 storage[i] = opt[j];
139 }
140 storage[i] = ' ';
141 ++i;
142 }
143 length = i;
144 if (i > 0U) {
145 storage[i - 1] = '\0';
146 }
147 }
148
149 constexpr operator ConstStringRef() const {
150 return ConstStringRef(storage, (length > 0U) ? (length - 1) : 0U);
151 }
152
153 constexpr operator const char *() const {
154 return storage;
155 }
156
157 protected:
158 char storage[MaxLength + nullterminateSize] = {};
159 size_t length = 0U;
160};

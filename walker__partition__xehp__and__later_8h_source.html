57template <typename Command>
58Command *putCommand(void *&inputAddress, uint32_t &totalBytesProgrammed) {
59 totalBytesProgrammed += sizeof(Command);
60 auto commandToReturn = reinterpret_cast<Command *>(inputAddress);
61 inputAddress = ptrOffset(inputAddress, sizeof(Command));
62 return commandToReturn;
63}

65template <typename GfxFamily>
66uint32_t computePartitionCountAndPartitionType(uint32_t preferredMinimalPartitionCount,
67 bool preferStaticPartitioning,
68 const Vec3<size_t> &groupStart,
69 const Vec3<size_t> &groupCount,
70 std::optional<typename COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE> requestedPartitionType,
71 typename COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE *outSelectedPartitionType,
72 bool *outSelectStaticPartitioning) {
73 // For non uniform starting point, there is no support for partition in Hardware. Disable partitioning and select dynamic algorithm
74 if (groupStart.x || groupStart.y || groupStart.z) {
75 *outSelectedPartitionType = COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_DISABLED;
76 *outSelectStaticPartitioning = false;
77 return 1u;
78 }
79
80 size_t workgroupCount = 0u;
81 bool disablePartitionForPartitionCountOne{};
82
83 if (NEO::DebugManager.flags.ExperimentalSetWalkerPartitionType.get() != -1) {
84 requestedPartitionType = static_cast<typename COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE>(NEO::DebugManager.flags.ExperimentalSetWalkerPartitionType.get());
85 }
86
87 if (requestedPartitionType.has_value()) {
88 switch (requestedPartitionType.value()) {
89 case COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_X:
90 workgroupCount = groupCount.x;
91 break;
92 case COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_Y:
93 workgroupCount = groupCount.y;
94 break;
95 case COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_Z:
96 workgroupCount = groupCount.z;
97 break;
98 default:
99 UNRECOVERABLE_IF(true);
100 }
101 *outSelectedPartitionType = requestedPartitionType.value();
102 disablePartitionForPartitionCountOne = false;
103 } else {
104 const size_t maxDimension = std::max({groupCount.z, groupCount.y, groupCount.x});
105
106 auto goWithMaxAlgorithm = !preferStaticPartitioning;
107 if (NEO::DebugManager.flags.WalkerPartitionPreferHighestDimension.get() != -1) {
108 goWithMaxAlgorithm = !!!NEO::DebugManager.flags.WalkerPartitionPreferHighestDimension.get();
109 }
110
111 //compute misaligned %, accept imbalance below threshold in favor of Z/Y/X distribution.
112 const float minimalThreshold = 0.05f;
113 float zImbalance = static_cast<float>(groupCount.z - alignDown(groupCount.z, preferredMinimalPartitionCount)) / static_cast<float>(groupCount.z);
114 float yImbalance = static_cast<float>(groupCount.y - alignDown(groupCount.y, preferredMinimalPartitionCount)) / static_cast<float>(groupCount.y);
115
116 //we first try with deepest dimension to see if we can partition there
117 if (groupCount.z > 1 && (zImbalance <= minimalThreshold)) {
118 *outSelectedPartitionType = COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_Z;
119 } else if (groupCount.y > 1 && (yImbalance < minimalThreshold)) {
120 *outSelectedPartitionType = COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_Y;
121 } else if (groupCount.x % preferredMinimalPartitionCount == 0) {
122 *outSelectedPartitionType = COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_X;
123 }
124 //if we are here then there is no dimension that results in even distribution, choose max dimension to minimize impact
125 else {
126 goWithMaxAlgorithm = true;
127 }
128
129 if (goWithMaxAlgorithm) {
130 // default mode, select greatest dimension
131 if (maxDimension == groupCount.x) {
132 *outSelectedPartitionType = COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_X;
133 } else if (maxDimension == groupCount.y) {
134 *outSelectedPartitionType = COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_Y;
135 } else {
136 *outSelectedPartitionType = COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_Z;
137 }
138 }
139
140 workgroupCount = maxDimension;
141 disablePartitionForPartitionCountOne = true;
142 }
143
144 // Static partitioning - partition count == tile count
145 *outSelectStaticPartitioning = preferStaticPartitioning;
146 if (preferStaticPartitioning) {
147 return preferredMinimalPartitionCount;
148 }
149
150 // Dynamic partitioning - compute optimal partition count
151 size_t partitionCount = std::min(static_cast<size_t>(16u), workgroupCount);
152 partitionCount = Math::prevPowerOfTwo(partitionCount);
153 if (NEO::DebugManager.flags.SetMinimalPartitionSize.get() != 0) {
154 const auto workgroupPerPartitionThreshold = NEO::DebugManager.flags.SetMinimalPartitionSize.get() == -1
155 ? 512u
156 : static_cast<unsigned>(NEO::DebugManager.flags.SetMinimalPartitionSize.get());
157 preferredMinimalPartitionCount = std::max(2u, preferredMinimalPartitionCount);
158
159 while (partitionCount > preferredMinimalPartitionCount) {
160 auto workgroupsPerPartition = workgroupCount / partitionCount;
161 if (workgroupsPerPartition >= workgroupPerPartitionThreshold) {
162 break;
163 }
164 partitionCount = partitionCount / 2;
165 }
166 }
167
168 if (partitionCount == 1u && disablePartitionForPartitionCountOne) {
169 *outSelectedPartitionType = COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_DISABLED;
170 }
171
172 return static_cast<uint32_t>(partitionCount);
173}

175template <typename GfxFamily>
176uint32_t computePartitionCountAndSetPartitionType(COMPUTE_WALKER<GfxFamily> *walker,
177 uint32_t preferredMinimalPartitionCount,
178 bool preferStaticPartitioning,
179 bool usesImages,
180 bool *outSelectStaticPartitioning) {
181 const Vec3<size_t> groupStart = {walker->getThreadGroupIdStartingX(), walker->getThreadGroupIdStartingY(), walker->getThreadGroupIdStartingZ()};
182 const Vec3<size_t> groupCount = {walker->getThreadGroupIdXDimension(), walker->getThreadGroupIdYDimension(), walker->getThreadGroupIdZDimension()};
183 std::optional<typename COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE> requestedPartitionType{};
184 if (usesImages) {
185 requestedPartitionType = COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_X;
186 }
187 typename COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE partitionType{};
188 const auto partitionCount = computePartitionCountAndPartitionType<GfxFamily>(preferredMinimalPartitionCount,
189 preferStaticPartitioning,
190 groupStart,
191 groupCount,
192 requestedPartitionType,
193 &partitionType,
194 outSelectStaticPartitioning);
195 walker->setPartitionType(partitionType);
196 return partitionCount;
197}

222template <typename GfxFamily>
223bool programWparidMask(void *&inputAddress, uint32_t &totalBytesProgrammed, uint32_t partitionCount) {
224 //currently only power of 2 values of partitionCount are being supported
225 if (!Math::isPow2(partitionCount) || partitionCount > 16) {
226 return false;
227 }
228
229 auto mask = 0xFFE0;
230 auto fillValue = 0x10;
231 auto count = partitionCount;
232 while (count < 16) {
233 fillValue |= (fillValue >> 1);
234 count *= 2;
235 }
236 mask |= (mask | fillValue);
237
238 programRegisterWithValue<GfxFamily>(inputAddress, predicationMaskCCSOffset, totalBytesProgrammed, mask);
239 return true;
240}

385template <typename GfxFamily>
386void programSelfCleanupEndSection(void *&inputAddress,
387 uint32_t &totalBytesProgrammed,
388 uint64_t finalSyncTileCountAddress,
389 uint64_t baseAddressForCleanup,
390 size_t fieldsForCleanupCount,
391 uint32_t tileCount,
392 bool useAtomicsForSelfCleanup) {
393 // Synchronize tiles, so the fields are not cleared while still in use
394 programTilesSynchronizationWithAtomics<GfxFamily>(inputAddress, totalBytesProgrammed, finalSyncTileCountAddress, tileCount);
395
396 for (auto fieldIndex = 0u; fieldIndex < fieldsForCleanupCount; fieldIndex++) {
397 const uint64_t addressForCleanup = baseAddressForCleanup + fieldIndex * sizeof(uint32_t);
398 programSelfCleanupSection<GfxFamily>(inputAddress,
399 totalBytesProgrammed,
400 addressForCleanup,
401 useAtomicsForSelfCleanup);
402 }
403
404 //this synchronization point ensures that all tiles finished zeroing and will fairly access control section atomic variables
405 programTilesSynchronizationWithAtomics<GfxFamily>(inputAddress, totalBytesProgrammed, finalSyncTileCountAddress, 2 * tileCount);
406}

425template <typename GfxFamily>
426uint64_t computeControlSectionOffset(WalkerPartitionArgs &args) {
427 uint64_t size = 0u;
428
429 size += args.synchronizeBeforeExecution ? computeTilesSynchronizationWithAtomicsSectionSize<GfxFamily>() : 0;
430 size += sizeof(LOAD_REGISTER_IMM<GfxFamily>); //predication mask
431 size += sizeof(MI_ATOMIC<GfxFamily>); //current id for partition
432 size += sizeof(LOAD_REGISTER_REG<GfxFamily>); //id into register
433 size += sizeof(MI_SET_PREDICATE<GfxFamily>) * 2 +
434 sizeof(BATCH_BUFFER_START<GfxFamily>) * 2;
435 size += (args.semaphoreProgrammingRequired ? sizeof(MI_SEMAPHORE_WAIT<GfxFamily>) * args.partitionCount : 0u);
436 size += computeWalkerSectionSize<GfxFamily>();
437 size += args.emitPipeControlStall ? sizeof(PIPE_CONTROL<GfxFamily>) : 0u;
438 if (args.crossTileAtomicSynchronization || args.emitSelfCleanup) {
439 size += computeTilesSynchronizationWithAtomicsSectionSize<GfxFamily>();
440 }
441 if (args.emitSelfCleanup) {
442 size += computeSelfCleanupSectionSize<GfxFamily>(args.useAtomicsForSelfCleanup);
443 }
444 size += args.preferredStaticPartitioning ? sizeof(LOAD_REGISTER_MEM<GfxFamily>) : 0u;
445 return size;
446}

450 return computeControlSectionOffset<GfxFamily>(args) -
451 computeWalkerSectionSize<GfxFamily>();

454template <typename GfxFamily>
455void programPartitionedWalker(void *&inputAddress, uint32_t &totalBytesProgrammed,
456 COMPUTE_WALKER<GfxFamily> *inputWalker,
457 uint32_t partitionCount) {
458 auto computeWalker = putCommand<COMPUTE_WALKER<GfxFamily>>(inputAddress, totalBytesProgrammed);
459 COMPUTE_WALKER<GfxFamily> cmd = *inputWalker;
460
461 if (partitionCount > 1) {
462 auto partitionType = inputWalker->getPartitionType();
463
464 assert(inputWalker->getThreadGroupIdStartingX() == 0u);
465 assert(inputWalker->getThreadGroupIdStartingY() == 0u);
466 assert(inputWalker->getThreadGroupIdStartingZ() == 0u);
467 assert(partitionType != COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_DISABLED);
468
469 cmd.setWorkloadPartitionEnable(true);
470
471 auto workgroupCount = 0u;
472 if (partitionType == COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_X) {
473 workgroupCount = inputWalker->getThreadGroupIdXDimension();
474 } else if (partitionType == COMPUTE_WALKER<GfxFamily>::PARTITION_TYPE::PARTITION_TYPE_Y) {
475 workgroupCount = inputWalker->getThreadGroupIdYDimension();
476 } else {
477 workgroupCount = inputWalker->getThreadGroupIdZDimension();
478 }
479
480 cmd.setPartitionSize((workgroupCount + partitionCount - 1u) / partitionCount);
481 }
482 *computeWalker = cmd;
483}

485/* SAMPLE COMMAND BUFFER STRUCTURE, birds eye view for 16 partitions, 4 tiles
486//inital setup section
4871. MI_LOAD_REGISTER(PREDICATION_MASK, active partition mask )
488//loop 1 - loop as long as there are partitions to be serviced
4892. MI_ATOMIC_INC( ATOMIC LOCATION #31 within CMD buffer )
4903. MI_LOAD_REGISTER_REG ( ATOMIC RESULT -> WPARID )
4914. MI_SET_PREDICATE( WPARID MODE )
4925. BATCH_BUFFER_START( LOCATION #28 ) // this will not be executed if partition outside of active virtual partitions
493//loop 1 ends here, if we are here it means there are no more partitions
4946. MI_SET_PREDICATE ( OFF )
495//Walker synchronization section starts here, make sure that Walker is done
4967, PIPE_CONTROL ( DC_FLUSH )
497//wait for all post syncs to make sure whole work is done, caller needs to set them to 1.
498//now epilogue starts synchro all engines prior to coming back to RING, this will be once per command buffer to make sure that all engines actually passed via cmd buffer.
499//epilogue section, make sure every tile completed prior to continuing
500//This is cross-tile synchronization
50124. ATOMIC_INC( LOCATION #31)
50225. WAIT_FOR_SEMAPHORE ( LOCATION #31, LOWER THEN 4 ) // wait till all tiles hit atomic
50326. PIPE_CONTROL ( TAG UPDATE ) (not implemented)
50427. BATCH_BUFFER_STAT (LOCATION #32) // go to the very end
505//Walker section
50628. COMPUTE_WALKER
50729. BATCH BUFFER_START ( GO BACK TO #2)
508//Batch Buffer Control Data section, there are no real commands here but we have memory here
509//That will be updated via atomic operations.
51030. uint32_t virtualPartitionID //atomic location
51131. uint32_t completionTileID //all tiles needs to report completion
51232. BATCH_BUFFER_END ( optional )
513*/
514
515template <typename GfxFamily>
516void constructDynamicallyPartitionedCommandBuffer(void *cpuPointer,
517 uint64_t gpuAddressOfAllocation,
518 COMPUTE_WALKER<GfxFamily> *inputWalker,
519 uint32_t &totalBytesProgrammed,
520 WalkerPartitionArgs &args,
521 const NEO::HardwareInfo &hwInfo) {
522 totalBytesProgrammed = 0u;
523 void *currentBatchBufferPointer = cpuPointer;
524
525 auto controlSectionOffset = computeControlSectionOffset<GfxFamily>(args);
526 if (args.synchronizeBeforeExecution) {
527 auto tileAtomicAddress = gpuAddressOfAllocation + controlSectionOffset + offsetof(BatchBufferControlData, inTileCount);
528 programTilesSynchronizationWithAtomics<GfxFamily>(currentBatchBufferPointer, totalBytesProgrammed, tileAtomicAddress, args.tileCount);
529 }
530
531 programWparidMask<GfxFamily>(currentBatchBufferPointer, totalBytesProgrammed, args.partitionCount);
532
533 programMiAtomic<GfxFamily>(currentBatchBufferPointer,

625template <typename GfxFamily>
626uint64_t computeStaticPartitioningControlSectionOffset(WalkerPartitionArgs &args) {
627 const auto beforeExecutionSyncAtomicSize = args.synchronizeBeforeExecution
628 ? computeTilesSynchronizationWithAtomicsSectionSize<GfxFamily>()
629 : 0u;
630 const auto afterExecutionSyncAtomicSize = (args.crossTileAtomicSynchronization || args.emitSelfCleanup)
631 ? computeTilesSynchronizationWithAtomicsSectionSize<GfxFamily>()
632 : 0u;
633 const auto afterExecutionSyncPostSyncSize = args.semaphoreProgrammingRequired
634 ? sizeof(MI_SEMAPHORE_WAIT<GfxFamily>) * args.partitionCount
635 : 0u;
636 const auto selfCleanupSectionSize = args.emitSelfCleanup
637 ? computeSelfCleanupSectionSize<GfxFamily>(args.useAtomicsForSelfCleanup)
638 : 0u;
639 const auto wparidRegisterSize = args.initializeWparidRegister
640 ? sizeof(LOAD_REGISTER_MEM<GfxFamily>)
641 : 0u;
642 const auto pipeControlSize = args.emitPipeControlStall
643 ? sizeof(PIPE_CONTROL<GfxFamily>)
644 : 0u;
645 const auto bbStartSize = isStartAndControlSectionRequired<GfxFamily>(args)
646 ? sizeof(BATCH_BUFFER_START<GfxFamily>)
647 : 0u;
648 return beforeExecutionSyncAtomicSize +
649 wparidRegisterSize +
650 pipeControlSize +
651 sizeof(COMPUTE_WALKER<GfxFamily>) +
652 selfCleanupSectionSize +
653 afterExecutionSyncAtomicSize +
654 afterExecutionSyncPostSyncSize +
655 bbStartSize;
656}

658template <typename GfxFamily>
659void constructStaticallyPartitionedCommandBuffer(void *cpuPointer,
660 uint64_t gpuAddressOfAllocation,
661 COMPUTE_WALKER<GfxFamily> *inputWalker,
662 uint32_t &totalBytesProgrammed,
663 WalkerPartitionArgs &args,
664 const NEO::HardwareInfo &hwInfo) {
665 totalBytesProgrammed = 0u;
666 void *currentBatchBufferPointer = cpuPointer;
667
668 // Get address of the control section
669 const auto controlSectionOffset = computeStaticPartitioningControlSectionOffset<GfxFamily>(args);
670 const auto afterControlSectionOffset = controlSectionOffset + sizeof(StaticPartitioningControlSection);
671
672 // Synchronize tiles before walker
673 if (args.synchronizeBeforeExecution) {
674 const auto atomicAddress = gpuAddressOfAllocation + controlSectionOffset + offsetof(StaticPartitioningControlSection, synchronizeBeforeWalkerCounter);
675 programTilesSynchronizationWithAtomics<GfxFamily>(currentBatchBufferPointer, totalBytesProgrammed, atomicAddress, args.tileCount);
676 }
677
678 // Load partition ID to wparid register and execute walker
679 if (args.initializeWparidRegister) {
680 programMiLoadRegisterMem<GfxFamily>(currentBatchBufferPointer, totalBytesProgrammed, args.workPartitionAllocationGpuVa, wparidCCSOffset);
681 }
682 programPartitionedWalker<GfxFamily>(currentBatchBufferPointer, totalBytesProgrammed, inputWalker, args.partitionCount);
683
684 // Prepare for cleanup section
685 if (args.emitSelfCleanup) {
686 const auto finalSyncTileCountField = gpuAddressOfAllocation + controlSectionOffset + offsetof(StaticPartitioningControlSection, finalSyncTileCounter);
687 programSelfCleanupSection<GfxFamily>(currentBatchBufferPointer, totalBytesProgrammed, finalSyncTileCountField, args.useAtomicsForSelfCleanup);
688 }
689
690 if (args.emitPipeControlStall) {
691 NEO::PipeControlArgs args;
692 args.dcFlushEnable = NEO::MemorySynchronizationCommands<GfxFamily>::getDcFlushEnable(true, hwInfo);
693 programPipeControlCommand<GfxFamily>(currentBatchBufferPointer, totalBytesProgrammed, args);
694 }
695
696 // Synchronize tiles after walker
697 if (args.semaphoreProgrammingRequired) {
698 programTilesSynchronizationWithPostSyncs<GfxFamily>(currentBatchBufferPointer, totalBytesProgrammed, inputWalker, args.partitionCount);
699 }
700
701 if (args.crossTileAtomicSynchronization || args.emitSelfCleanup) {
702 const auto atomicAddress = gpuAddressOfAllocation + controlSectionOffset + offsetof(StaticPartitioningControlSection, synchronizeAfterWalkerCounter);
703 programTilesSynchronizationWithAtomics<GfxFamily>(currentBatchBufferPointer, totalBytesProgrammed, atomicAddress, args.tileCount);
704 }
705
706 // Jump over the control section only when needed
707 if (isStartAndControlSectionRequired<GfxFamily>(args)) {
708 programMiBatchBufferStart<GfxFamily>(currentBatchBufferPointer, totalBytesProgrammed, gpuAddressOfAllocation + afterControlSectionOffset, false, args.secondaryBatchBuffer);
709
710 // Control section
711 DEBUG_BREAK_IF(totalBytesProgrammed != controlSectionOffset);
712 StaticPartitioningControlSection *controlSection = putCommand<StaticPartitioningControlSection>(currentBatchBufferPointer, totalBytesProgrammed);
713 controlSection->synchronizeBeforeWalkerCounter = 0u;
714 controlSection->synchronizeAfterWalkerCounter = 0u;
715 controlSection->finalSyncTileCounter = 0u;
716 DEBUG_BREAK_IF(totalBytesProgrammed != afterControlSectionOffset);
717 }
718
719 // Cleanup section
720 if (args.emitSelfCleanup) {
721 const auto finalSyncTileCountAddress = gpuAddressOfAllocation + controlSectionOffset + offsetof(StaticPartitioningControlSection, finalSyncTileCounter);
722 programSelfCleanupEndSection<GfxFamily>(currentBatchBufferPointer,
723 totalBytesProgrammed,
724 finalSyncTileCountAddress,
725 gpuAddressOfAllocation + controlSectionOffset,
726 staticPartitioningFieldsForCleanupCount,
727 args.tileCount,
728 args.useAtomicsForSelfCleanup);
729 }

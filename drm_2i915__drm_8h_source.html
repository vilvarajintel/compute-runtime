134/*
135 * MOCS indexes used for GPU surfaces, defining the cacheability of the
136 * surface data and the coherency for this data wrt. CPU vs. GPU accesses.
137 */
138enum i915_mocs_table_index {
139 /*
140 * Not cached anywhere, coherency between CPU and GPU accesses is
141 * guaranteed.
142 */
143 I915_MOCS_UNCACHED,
144 /*
145 * Cacheability and coherency controlled by the kernel automatically
146 * based on the DRM_I915_GEM_SET_CACHING IOCTL setting and the current
147 * usage of the surface (used for display scanout or not).
148 */
149 I915_MOCS_PTE,
150 /*
151 * Cached in all GPU caches available on the platform.
152 * Coherency between CPU and GPU accesses to the surface is not
153 * guaranteed without extra synchronization.
154 */
155 I915_MOCS_CACHED,
156};

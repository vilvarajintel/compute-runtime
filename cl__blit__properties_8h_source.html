18struct ClBlitProperties {
19 static BlitProperties constructProperties(BlitterConstants::BlitDirection blitDirection,
20 CommandStreamReceiver &commandStreamReceiver,
21 const BuiltinOpParams &builtinOpParams) {
22
23 auto rootDeviceIndex = commandStreamReceiver.getRootDeviceIndex();
24 auto clearColorAllocation = commandStreamReceiver.getClearColorAllocation();
25 BlitProperties blitProperties{};
26
27 if (BlitterConstants::BlitDirection::BufferToBuffer == blitDirection ||
28 BlitterConstants::BlitDirection::ImageToImage == blitDirection) {
29 auto dstOffset = builtinOpParams.dstOffset.x;
30 auto srcOffset = builtinOpParams.srcOffset.x;
31 GraphicsAllocation *dstAllocation = nullptr;
32 GraphicsAllocation *srcAllocation = nullptr;
33
34 if (!builtinOpParams.dstSvmAlloc) {
35 dstOffset += builtinOpParams.dstMemObj->getOffset();
36 srcOffset += builtinOpParams.srcMemObj->getOffset();
37 dstAllocation = builtinOpParams.dstMemObj->getGraphicsAllocation(rootDeviceIndex);
38 srcAllocation = builtinOpParams.srcMemObj->getGraphicsAllocation(rootDeviceIndex);
39 } else {
40 dstAllocation = builtinOpParams.dstSvmAlloc;
41 srcAllocation = builtinOpParams.srcSvmAlloc;
42 dstOffset += ptrDiff(builtinOpParams.dstPtr, dstAllocation->getGpuAddress());
43 srcOffset += ptrDiff(builtinOpParams.srcPtr, srcAllocation->getGpuAddress());
44 }
45
46 blitProperties = BlitProperties::constructPropertiesForCopy(dstAllocation,
47 srcAllocation,
48 {dstOffset, builtinOpParams.dstOffset.y, builtinOpParams.dstOffset.z},
49 {srcOffset, builtinOpParams.srcOffset.y, builtinOpParams.srcOffset.z},
50 builtinOpParams.size,
51 builtinOpParams.srcRowPitch, builtinOpParams.srcSlicePitch,
52 builtinOpParams.dstRowPitch, builtinOpParams.dstSlicePitch, clearColorAllocation);
53
54 if (BlitterConstants::BlitDirection::ImageToImage == blitDirection) {
55 blitProperties.blitDirection = blitDirection;
56 setBlitPropertiesForImage(blitProperties, builtinOpParams);
57 }
58 return blitProperties;
59 }
60
61 GraphicsAllocation *gpuAllocation = nullptr;
62 Vec3<size_t> copyOffset = 0;
63
64 void *hostPtr = nullptr;
65 Vec3<size_t> hostPtrOffset = 0;
66
67 uint64_t memObjGpuVa = 0;
68 uint64_t hostAllocGpuVa = 0;
69
70 GraphicsAllocation *hostAllocation = builtinOpParams.transferAllocation;
71
72 Vec3<size_t> copySize = 0;
73 size_t hostRowPitch = 0;
74 size_t hostSlicePitch = 0;
75 size_t gpuRowPitch = 0;
76 size_t gpuSlicePitch = 0;
77
78 if (BlitterConstants::BlitDirection::HostPtrToBuffer == blitDirection ||
79 BlitterConstants::BlitDirection::HostPtrToImage == blitDirection) {
80 // write buffer/image
81 hostPtr = builtinOpParams.srcPtr;
82 hostPtrOffset = builtinOpParams.srcOffset;
83 copyOffset = builtinOpParams.dstOffset;
84
85 memObjGpuVa = castToUint64(builtinOpParams.dstPtr);
86 hostAllocGpuVa = castToUint64(builtinOpParams.srcPtr);
87
88 if (builtinOpParams.dstSvmAlloc) {
89 gpuAllocation = builtinOpParams.dstSvmAlloc;
90 hostAllocation = builtinOpParams.srcSvmAlloc;
91 } else {
92 gpuAllocation = builtinOpParams.dstMemObj->getGraphicsAllocation(rootDeviceIndex);
93 memObjGpuVa = (gpuAllocation->getGpuAddress() + builtinOpParams.dstMemObj->getOffset());
94 }
95
96 hostRowPitch = builtinOpParams.srcRowPitch;
97 hostSlicePitch = builtinOpParams.srcSlicePitch;
98 gpuRowPitch = builtinOpParams.dstRowPitch;
99 gpuSlicePitch = builtinOpParams.dstSlicePitch;
100 copySize = builtinOpParams.size;
101 }
102
103 if (BlitterConstants::BlitDirection::BufferToHostPtr == blitDirection ||
104 BlitterConstants::BlitDirection::ImageToHostPtr == blitDirection) {
105 // read buffer/image
106 hostPtr = builtinOpParams.dstPtr;
107 hostPtrOffset = builtinOpParams.dstOffset;
108 copyOffset = builtinOpParams.srcOffset;
109
110 memObjGpuVa = castToUint64(builtinOpParams.srcPtr);
111 hostAllocGpuVa = castToUint64(builtinOpParams.dstPtr);
112
113 if (builtinOpParams.srcSvmAlloc) {
114 gpuAllocation = builtinOpParams.srcSvmAlloc;
115 hostAllocation = builtinOpParams.dstSvmAlloc;
116 } else {
117 gpuAllocation = builtinOpParams.srcMemObj->getGraphicsAllocation(rootDeviceIndex);
118 memObjGpuVa = (gpuAllocation->getGpuAddress() + builtinOpParams.srcMemObj->getOffset());
119 }
120
121 hostRowPitch = builtinOpParams.dstRowPitch;
122 hostSlicePitch = builtinOpParams.dstSlicePitch;
123 gpuRowPitch = builtinOpParams.srcRowPitch;
124 gpuSlicePitch = builtinOpParams.srcSlicePitch;
125 copySize = builtinOpParams.size;
126 }
127
128 UNRECOVERABLE_IF(BlitterConstants::BlitDirection::HostPtrToBuffer != blitDirection &&
129 BlitterConstants::BlitDirection::BufferToHostPtr != blitDirection &&
130 BlitterConstants::BlitDirection::HostPtrToImage != blitDirection &&
131 BlitterConstants::BlitDirection::ImageToHostPtr != blitDirection);
132
133 blitProperties = BlitProperties::constructPropertiesForReadWrite(blitDirection, commandStreamReceiver, gpuAllocation,
134 hostAllocation, hostPtr, memObjGpuVa, hostAllocGpuVa,
135 hostPtrOffset, copyOffset, copySize,
136 hostRowPitch, hostSlicePitch,
137 gpuRowPitch, gpuSlicePitch);
138
139 if (BlitterConstants::BlitDirection::HostPtrToImage == blitDirection ||
140 BlitterConstants::BlitDirection::ImageToHostPtr == blitDirection) {
141 setBlitPropertiesForImage(blitProperties, builtinOpParams);
142 }
143
144 return blitProperties;
145 }

147 static BlitterConstants::BlitDirection obtainBlitDirection(uint32_t commandType) {
148
149 switch (commandType) {
150 case CL_COMMAND_WRITE_BUFFER:
151 case CL_COMMAND_WRITE_BUFFER_RECT:
152 return BlitterConstants::BlitDirection::HostPtrToBuffer;
153 case CL_COMMAND_READ_BUFFER:
154 case CL_COMMAND_READ_BUFFER_RECT:
155 return BlitterConstants::BlitDirection::BufferToHostPtr;
156 case CL_COMMAND_COPY_BUFFER:
157 case CL_COMMAND_COPY_BUFFER_RECT:
158 case CL_COMMAND_SVM_MEMCPY:
159 return BlitterConstants::BlitDirection::BufferToBuffer;
160 case CL_COMMAND_WRITE_IMAGE:
161 return BlitterConstants::BlitDirection::HostPtrToImage;
162 case CL_COMMAND_READ_IMAGE:
163 return BlitterConstants::BlitDirection::ImageToHostPtr;
164 case CL_COMMAND_COPY_IMAGE:
165 return BlitterConstants::BlitDirection::ImageToImage;
166 default:
167 UNRECOVERABLE_IF(true);
168 }
169 }

171 static void adjustBlitPropertiesForImage(MemObj *memObj, Vec3<size_t> &size, size_t &bytesPerPixel, uint64_t &gpuAddress, size_t &rowPitch, size_t &slicePitch) {
172 auto image = castToObject<Image>(memObj);
173 const auto &imageDesc = image->getImageDesc();
174 auto image_width = imageDesc.image_width;
175 auto image_height = imageDesc.image_height;
176 auto image_depth = imageDesc.image_depth;
177
178 if (imageDesc.image_type == CL_MEM_OBJECT_IMAGE2D_ARRAY) {
179 image_depth = std::max(image_depth, imageDesc.image_array_size);
180 }
181
182 SurfaceOffsets surfaceOffsets;
183 image->getSurfaceOffsets(surfaceOffsets);
184 gpuAddress += surfaceOffsets.offset;
185 size.x = image_width;
186 size.y = image_height ? image_height : 1;
187 size.z = image_depth ? image_depth : 1;
188 bytesPerPixel = image->getSurfaceFormatInfo().surfaceFormat.ImageElementSizeInBytes;
189 rowPitch = imageDesc.image_row_pitch;
190 slicePitch = imageDesc.image_slice_pitch;
191 }
192
193 static void setBlitPropertiesForImage(BlitProperties &blitProperties, const BuiltinOpParams &builtinOpParams) {
194 size_t srcRowPitch = builtinOpParams.srcRowPitch;
195 size_t dstRowPitch = builtinOpParams.dstRowPitch;
196 size_t srcSlicePitch = builtinOpParams.srcSlicePitch;
197 size_t dstSlicePitch = builtinOpParams.dstSlicePitch;
198
199 if (blitProperties.blitDirection == BlitterConstants::BlitDirection::ImageToHostPtr ||
200 blitProperties.blitDirection == BlitterConstants::BlitDirection::ImageToImage) {
201 adjustBlitPropertiesForImage(builtinOpParams.srcMemObj, blitProperties.srcSize, blitProperties.bytesPerPixel,
202 blitProperties.srcGpuAddress, srcRowPitch, srcSlicePitch);
203 }
204
205 if (blitProperties.blitDirection == BlitterConstants::BlitDirection::HostPtrToImage ||
206 blitProperties.blitDirection == BlitterConstants::BlitDirection::ImageToImage) {
207 adjustBlitPropertiesForImage(builtinOpParams.dstMemObj, blitProperties.dstSize, blitProperties.bytesPerPixel,
208 blitProperties.dstGpuAddress, dstRowPitch, dstSlicePitch);
209 }
210
211 blitProperties.srcRowPitch = srcRowPitch ? srcRowPitch : blitProperties.srcSize.x * blitProperties.bytesPerPixel;
212 blitProperties.dstRowPitch = dstRowPitch ? dstRowPitch : blitProperties.dstSize.x * blitProperties.bytesPerPixel;
213 blitProperties.srcSlicePitch = srcSlicePitch ? srcSlicePitch : blitProperties.srcSize.y * blitProperties.srcRowPitch;
214 blitProperties.dstSlicePitch = dstSlicePitch ? dstSlicePitch : blitProperties.dstSize.y * blitProperties.dstRowPitch;
215 }
216};

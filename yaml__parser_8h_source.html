57inline bool isSpecificNameIdentifier(ConstStringRef wholeText, const char *parsePos, ConstStringRef pattern) {
58 UNRECOVERABLE_IF(parsePos < wholeText.begin());
59 bool hasEnoughText = (reinterpret_cast<uintptr_t>(parsePos) + pattern.size() <= reinterpret_cast<uintptr_t>(wholeText.end()));
60 bool isEnd = (reinterpret_cast<uintptr_t>(parsePos) + pattern.size() == reinterpret_cast<uintptr_t>(wholeText.end()));
61 bool matched = hasEnoughText &&
62 ((pattern == ConstStringRef(parsePos, pattern.size())) & (isEnd || (false == isNameIdentifierCharacter(parsePos[pattern.size()]))));
63 return matched;
64}

72constexpr const char *consumeNumberOrSign(ConstStringRef wholeText, const char *parsePos, bool allowSign = true) {
73 UNRECOVERABLE_IF(parsePos < wholeText.begin());
74 UNRECOVERABLE_IF(parsePos == wholeText.end());
75 auto parseEnd = wholeText.end();
76 if (isNumber(*parsePos)) {
77 auto it = parsePos + 1;
78 while (it < parseEnd) {
79 if (false == (isNumber(*it) | ('.' == *it))) {
80 break;
81 }
82 ++it;
83 }
84 if (it < parseEnd) {
85 return isLetter(*it) ? parsePos : it;
86 } else {
87 return it;
88 }
89 } else if (isSign(*parsePos) && allowSign) {
90 if (parsePos + 1 < wholeText.end()) {
91 return consumeNumberOrSign(wholeText, parsePos + 1, false);
92 } else {
93 return parsePos + 1;
94 }
95 }
96 return parsePos;
97}

99constexpr const char *consumeNameIdentifier(ConstStringRef wholeText, const char *parsePos) {
100 auto parseEnd = wholeText.end();
101 if (isNameIdentifierBeginningCharacter(*parsePos)) {
102 auto it = parsePos + 1;
103 while (it < parseEnd) {
104 if (false == isNameIdentifierCharacter(*it)) {
105 break;
106 }
107 ++it;
108 }
109 return it;
110 }
111 return parsePos;

114constexpr const char *consumeStringLiteral(ConstStringRef wholeText, const char *parsePos) {
115 auto stringLiteralBeg = *parsePos;
116 switch (stringLiteralBeg) {
117 default:
118 return parsePos;
119 case '\'':
120 break;
121 case '\"':
122 break;
123 }
124 auto parseEnd = wholeText.end();
125 auto it = parsePos + 1;
126 while (it < parseEnd) {
127 if (stringLiteralBeg == it[0]) {
128 if (it[-1] != '\\') { // allow escape characters
129 break;
130 }
131 }
132 ++it;
133 }
134 if (it == parseEnd) {
135 return parsePos; // unterminated literal
136 }
137 return it + 1;
138}

241template <typename T, typename It>
242inline bool reserveBasedOnEstimates(T &container, It beg, It end, It pos) {
243 if ((container.size() < container.capacity()) || (pos == beg)) {
244 return false;
245 }
246 DEBUG_BREAK_IF((beg > end) || (pos < beg));
247 auto normalizedPosInv = float(end - beg) / float(pos - beg);
248 auto estimatedTotalElements = static_cast<size_t>(container.size() * normalizedPosInv);
249 container.reserve(estimatedTotalElements);
250 return true;
251}

588template <>
589inline bool YamlParser::readValueChecked<bool>(const Node &node, bool &outValue) const {
590 if (invalidTokenId == node.value) {
591 return false;
592 }
593 const auto &token = tokens[node.value];
594 if (Token::Type::LiteralString != token.traits.type) {
595 return false;
596 }
597
598 // valid values : y/n yes/no true/false on/off (case insesitive)
599 if (token.len > 5) {
600 return false;
601 }
602
603 switch (token.traits.character0) {
604 default:
605 return false;
606 case 'y':
607 case 'Y': {
608 outValue = true;
609 switch (token.len) {
610 default:
611 return false;
612 case 1:
613 return true;
614 case 3:
615 return equalsCaseInsensitive(ConstStringRef("es"), ConstStringRef(token.cstrref().begin() + 1, 2));
616 }
617 break;
618 }
619 case 'n':
620 case 'N': {
621 outValue = false;
622 switch (token.len) {
623 default:
624 return false;
625 case 1:
626 return true;
627 case 2:
628 return ((token.cstrref()[1] == 'o') | (token.cstrref()[1] == 'O'));
629 }
630 break;
631 }
632 case 't':
633 case 'T': {
634 outValue = true;
635 if (token.len != 4) {
636 return false;
637 }
638 return equalsCaseInsensitive(ConstStringRef("rue"), ConstStringRef(token.cstrref().begin() + 1, 3));
639 }
640 case 'f':
641 case 'F': {
642 outValue = false;
643 if (token.len != 5) {
644 return false;
645 }
646 return equalsCaseInsensitive(ConstStringRef("alse"), ConstStringRef(token.cstrref().begin() + 1, 4));
647 }
648 case 'o':
649 case 'O': {
650 switch (token.len) {
651 default:
652 return false;
653 case 2:
654 outValue = true;
655 return ((token.cstrref()[1] == 'n') | (token.cstrref()[1] == 'N'));
656 case 3:
657 outValue = false;
658 return equalsCaseInsensitive(ConstStringRef("ff"), ConstStringRef(token.cstrref().begin() + 1, 2));
659 }
660 break;
661 }
662 }
663
664 return true;
665}

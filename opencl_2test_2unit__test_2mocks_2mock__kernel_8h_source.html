184 void setCrossThreadData(const void *crossThreadDataPattern, uint32_t newCrossThreadDataSize) {
185 if ((crossThreadData != nullptr) && (crossThreadData != mockCrossThreadData.data())) {
186 delete[] crossThreadData;
187 crossThreadData = nullptr;
188 crossThreadDataSize = 0;
189 }
190 if (crossThreadDataPattern && (newCrossThreadDataSize > 0)) {
191 mockCrossThreadData.clear();
192 mockCrossThreadData.insert(mockCrossThreadData.begin(), (char *)crossThreadDataPattern, ((char *)crossThreadDataPattern) + newCrossThreadDataSize);
193 } else {
194 mockCrossThreadData.resize(newCrossThreadDataSize, 0);
195 }
196
197 if (newCrossThreadDataSize == 0) {
198 crossThreadData = nullptr;
199 crossThreadDataSize = 0;
200 return;
201 }
202 crossThreadData = mockCrossThreadData.data();
203 crossThreadDataSize = static_cast<uint32_t>(mockCrossThreadData.size());

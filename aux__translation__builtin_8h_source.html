26 template <typename GfxFamily>
27 bool buildDispatchInfosForAuxTranslation(MultiDispatchInfo &multiDispatchInfo, const BuiltinOpParams &operationParams) const {
28 size_t kernelInstanceNumber = 0;
29 size_t numKernelObjectsToTranslate = multiDispatchInfo.getKernelObjsForAuxTranslation()->size();
30 resizeKernelInstances(numKernelObjectsToTranslate);
31 multiDispatchInfo.setBuiltinOpParams(operationParams);
32
33 for (auto &kernelObj : *multiDispatchInfo.getKernelObjsForAuxTranslation()) {
34 DispatchInfoBuilder<SplitDispatch::Dim::d1D, SplitDispatch::SplitMode::NoSplit> builder(clDevice);
35
36 UNRECOVERABLE_IF(builder.getMaxNumDispatches() != 1);
37
38 if (kernelInstanceNumber == 0) {
39 // Before Kernel
40 registerPipeControlProgramming<GfxFamily>(builder.getDispatchInfo(0).dispatchInitCommands, true);
41 }
42 if (kernelInstanceNumber == numKernelObjectsToTranslate - 1) {
43 // After Kernel
44 registerPipeControlProgramming<GfxFamily>(builder.getDispatchInfo(0).dispatchEpilogueCommands, false);
45 }
46
47 if (AuxTranslationDirection::AuxToNonAux == operationParams.auxTranslationDirection) {
48 builder.setKernel(convertToNonAuxKernel[kernelInstanceNumber++].get());
49 } else {
50 UNRECOVERABLE_IF(AuxTranslationDirection::NonAuxToAux != operationParams.auxTranslationDirection);
51 builder.setKernel(convertToAuxKernel[kernelInstanceNumber++].get());
52 }
53
54 size_t allocationSize = 0;
55 if (kernelObj.type == KernelObjForAuxTranslation::Type::MEM_OBJ) {
56 auto buffer = static_cast<Buffer *>(kernelObj.object);
57 builder.setArg(0, buffer);
58 builder.setArg(1, buffer);
59 allocationSize = alignUp(buffer->getSize(), 512);
60 } else {
61 DEBUG_BREAK_IF(kernelObj.type != KernelObjForAuxTranslation::Type::GFX_ALLOC);
62 auto svmAlloc = static_cast<GraphicsAllocation *>(kernelObj.object);
63 auto svmPtr = reinterpret_cast<void *>(svmAlloc->getGpuAddressToPatch());
64 builder.setArgSvmAlloc(0, svmPtr, svmAlloc);
65 builder.setArgSvmAlloc(1, svmPtr, svmAlloc);
66 allocationSize = alignUp(svmAlloc->getUnderlyingBufferSize(), 512);
67 }
68
69 size_t xGws = allocationSize / 16;
70
71 builder.setDispatchGeometry(Vec3<size_t>{xGws, 0, 0}, Vec3<size_t>{0, 0, 0}, Vec3<size_t>{0, 0, 0});
72 builder.bake(multiDispatchInfo);
73 }
74
75 return true;
76 }

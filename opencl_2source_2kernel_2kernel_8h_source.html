88 template <typename kernel_t = Kernel, typename program_t = Program>
89 static kernel_t *create(program_t *program, const KernelInfo &kernelInfo, ClDevice &clDevice, cl_int *errcodeRet) {
90 cl_int retVal;
91 kernel_t *pKernel = nullptr;
92
93 pKernel = new kernel_t(program, kernelInfo, clDevice);
94 retVal = pKernel->initialize();
95
96 if (retVal != CL_SUCCESS) {
97 delete pKernel;
98 pKernel = nullptr;
99 }
100
101 if (errcodeRet) {
102 *errcodeRet = retVal;
103 }
104
105 if (FileLoggerInstance().enabled()) {
106 std::string source;
107 program->getSource(source);
108 FileLoggerInstance().dumpKernel(kernelInfo.kernelDescriptor.kernelMetadata.kernelName, source);
109 }
110
111 return pKernel;
112 }

491 struct KernelConfig {
492 Vec3<size_t> gws;
493 Vec3<size_t> lws;

495 bool operator==(const KernelConfig &other) const { return this->gws == other.gws && this->lws == other.lws && this->offsets == other.offsets; }

497 struct KernelConfigHash {
498 size_t operator()(KernelConfig const &config) const {
499 auto hash = std::hash<size_t>{};
500 size_t gwsHashX = hash(config.gws.x);
501 size_t gwsHashY = hash(config.gws.y);
502 size_t gwsHashZ = hash(config.gws.z);
503 size_t gwsHash = hashCombine(gwsHashX, gwsHashY, gwsHashZ);
504 size_t lwsHashX = hash(config.lws.x);
505 size_t lwsHashY = hash(config.lws.y);
506 size_t lwsHashZ = hash(config.lws.z);
507 size_t lwsHash = hashCombine(lwsHashX, lwsHashY, lwsHashZ);
508 size_t offsetsHashX = hash(config.offsets.x);
509 size_t offsetsHashY = hash(config.offsets.y);
510 size_t offsetsHashZ = hash(config.offsets.z);
511 size_t offsetsHash = hashCombine(offsetsHashX, offsetsHashY, offsetsHashZ);
512 return hashCombine(gwsHash, lwsHash, offsetsHash);
513 }
514
515 size_t hashCombine(size_t hash1, size_t hash2, size_t hash3) const {
516 return (hash1 ^ (hash2 << 1u)) ^ (hash3 << 2u);
517 }

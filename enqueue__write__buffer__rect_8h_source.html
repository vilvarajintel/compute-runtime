21template <typename GfxFamily>
22cl_int CommandQueueHw<GfxFamily>::enqueueWriteBufferRect(
23 Buffer *buffer,
24 cl_bool blockingWrite,
25 const size_t *bufferOrigin,
26 const size_t *hostOrigin,
27 const size_t *region,
28 size_t bufferRowPitch,
29 size_t bufferSlicePitch,
30 size_t hostRowPitch,
31 size_t hostSlicePitch,
32 const void *ptr,
33 cl_uint numEventsInWaitList,
34 const cl_event *eventWaitList,
35 cl_event *event) {
36 const cl_command_type cmdType = CL_COMMAND_WRITE_BUFFER_RECT;
37
38 CsrSelectionArgs csrSelectionArgs{cmdType, buffer, {}, device->getRootDeviceIndex(), region};
39 CommandStreamReceiver &csr = selectCsrForBuiltinOperation(csrSelectionArgs);
40
41 auto isMemTransferNeeded = true;
42
43 if (buffer->isMemObjZeroCopy()) {
44 size_t bufferOffset;
45 size_t hostOffset;
46 computeOffsetsValueForRectCommands(&bufferOffset, &hostOffset, bufferOrigin, hostOrigin, region, bufferRowPitch, bufferSlicePitch, hostRowPitch, hostSlicePitch);
47 isMemTransferNeeded = buffer->checkIfMemoryTransferIsRequired(bufferOffset, hostOffset, ptr, cmdType);
48 }
49 if (!isMemTransferNeeded) {
50 return enqueueMarkerForReadWriteOperation(buffer, const_cast<void *>(ptr), cmdType, blockingWrite,
51 numEventsInWaitList, eventWaitList, event);
52 }
53
54 const size_t hostPtrSize = Buffer::calculateHostPtrSize(hostOrigin, region, hostRowPitch, hostSlicePitch);
55 const uint32_t rootDeviceIndex = getDevice().getRootDeviceIndex();
56 InternalMemoryType memoryType = InternalMemoryType::NOT_SPECIFIED;
57 GraphicsAllocation *mapAllocation = nullptr;
58 bool isCpuCopyAllowed = false;
59 getContext().tryGetExistingHostPtrAllocation(ptr, hostPtrSize, rootDeviceIndex, mapAllocation, memoryType, isCpuCopyAllowed);
60
61 auto eBuiltInOps = EBuiltInOps::CopyBufferRect;
62 if (forceStateless(buffer->getSize())) {
63 eBuiltInOps = EBuiltInOps::CopyBufferRectStateless;
64 }
65
66 void *srcPtr = const_cast<void *>(ptr);
67
68 MemObjSurface dstBufferSurf(buffer);
69 HostPtrSurface hostPtrSurf(srcPtr, hostPtrSize, true);
70 GeneralSurface mapSurface;
71 Surface *surfaces[] = {&dstBufferSurf, nullptr};
72
73 if (region[0] != 0 && region[1] != 0 && region[2] != 0) {
74 if (mapAllocation) {
75 surfaces[1] = &mapSurface;
76 mapSurface.setGraphicsAllocation(mapAllocation);
77 srcPtr = convertAddressWithOffsetToGpuVa(srcPtr, memoryType, *mapAllocation);
78 } else {
79 surfaces[1] = &hostPtrSurf;
80 bool status = csr.createAllocationForHostSurface(hostPtrSurf, false);
81 if (!status) {
82 return CL_OUT_OF_RESOURCES;
83 }
84 srcPtr = reinterpret_cast<void *>(hostPtrSurf.getAllocation()->getGpuAddress());
85 }
86 }
87
88 void *alignedSrcPtr = alignDown(srcPtr, 4);
89 size_t srcPtrOffset = ptrDiff(srcPtr, alignedSrcPtr);
90
91 BuiltinOpParams dc;
92 dc.srcPtr = alignedSrcPtr;
93 dc.dstMemObj = buffer;
94 dc.srcOffset = hostOrigin;
95 dc.srcOffset.x += srcPtrOffset;
96 dc.dstOffset = bufferOrigin;
97 dc.transferAllocation = hostPtrSurf.getAllocation();
98 dc.size = region;
99 dc.srcRowPitch = hostRowPitch;
100 dc.srcSlicePitch = hostSlicePitch;
101 dc.dstRowPitch = bufferRowPitch;
102 dc.dstSlicePitch = bufferSlicePitch;
103
104 MultiDispatchInfo dispatchInfo(dc);
105 dispatchBcsOrGpgpuEnqueue<CL_COMMAND_WRITE_BUFFER_RECT>(dispatchInfo, surfaces, eBuiltInOps, numEventsInWaitList, eventWaitList, event, blockingWrite, csr);
106
107 if (context->isProvidingPerformanceHints()) {
108 context->providePerformanceHint(CL_CONTEXT_DIAGNOSTICS_LEVEL_NEUTRAL_INTEL, CL_ENQUEUE_WRITE_BUFFER_RECT_REQUIRES_COPY_DATA, static_cast<cl_mem>(buffer));
109 }
110
111 return CL_SUCCESS;
112}
113} // namespace NEO

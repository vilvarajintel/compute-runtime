56 std::unique_ptr<BuiltinFunctionsLibImpl::BuiltinData> loadBuiltIn(NEO::EBuiltInOps::Type builtin, const char *builtInName) override {
57 using BuiltInCodeType = NEO::BuiltinCode::ECodeType;
58
59 auto builtInCodeType = NEO::DebugManager.flags.RebuildPrecompiledKernels.get() ? BuiltInCodeType::Intermediate : BuiltInCodeType::Binary;
60 auto builtInCode = builtInsLib->getBuiltinsLib().getBuiltinCode(builtin, builtInCodeType, *device->getNEODevice());
61
62 ze_result_t res;
63 std::unique_ptr<Module> module;
64 ze_module_handle_t moduleHandle;
65 ze_module_desc_t moduleDesc = {};
66 moduleDesc.format = builtInCode.type == BuiltInCodeType::Binary ? ZE_MODULE_FORMAT_NATIVE : ZE_MODULE_FORMAT_IL_SPIRV;
67 moduleDesc.pInputModule = reinterpret_cast<uint8_t *>(&builtInCode.resource[0]);
68 moduleDesc.inputSize = builtInCode.resource.size();
69 res = device->createModule(&moduleDesc, &moduleHandle, nullptr, ModuleType::Builtin);
70 UNRECOVERABLE_IF(res != ZE_RESULT_SUCCESS);
71
72 module.reset(Module::fromHandle(moduleHandle));
73
74 std::unique_ptr<Kernel> kernel;
75 ze_kernel_handle_t kernelHandle;
76 ze_kernel_desc_t kernelDesc = {};
77 kernelDesc.pKernelName = builtInName;
78 res = module->createKernel(&kernelDesc, &kernelHandle);
79 DEBUG_BREAK_IF(res != ZE_RESULT_SUCCESS);
80
81 kernel.reset(Kernel::fromHandle(kernelHandle));
82 return std::unique_ptr<BuiltinData>(new MockBuiltinDataTimestamp{std::move(module), std::move(kernel)});
83 }

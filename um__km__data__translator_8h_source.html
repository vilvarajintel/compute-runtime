53struct UmKmDataTempStorageBase {

63 void resize(size_t dynSize) {
64 auto oldSize = storage.size() * sizeof(uint64_t);
65 storage.resize((dynSize + sizeof(uint64_t) - 1) / sizeof(uint64_t));
66 requestedSize = dynSize;
67 memset(reinterpret_cast<char *>(data()) + oldSize, 0, storage.size() * sizeof(uint64_t) - oldSize);
68 }
69
70 size_t size() const {

72 }
73
74 protected:
75 static constexpr size_t staticSizeQwordsCount = (StaticSize + sizeof(uint64_t) - 1) / sizeof(uint64_t);
76 StackVec<uint64_t, staticSizeQwordsCount> storage;
77 size_t requestedSize = 0U;
78};
